80 subroutine check_redundant_v2d(mesg, u_comp, v_comp, G, is, ie, js, je, &
81  direction, stagger)
82  character(len=*), intent(in) :: mesg
83  real, dimension(NIMEMB_,NJMEM_), intent(in) :: u_comp
84  real, dimension(NIMEM_,NJMEMB_), intent(in) :: v_comp
85  type(ocean_grid_type), intent(inout) :: g
86  integer, optional, intent(in) :: is, ie, js, je

89 ! Arguments: u_comp - The u-component of the vector being checked.
90 ! (in) v_comp - The v-component of the vector being checked.
91 ! (in) mesg - A message indicating what is being checked.
92 ! (in) G - The ocean's grid structure.
93 ! (in/opt) is, ie, js, je - the i- and j- range of indices to check.
94 ! (in/opt) direction - the direction flag to be passed to pass_vector.
95 ! (in/opt) stagger - the stagger flag to be passed to pass_vector.
96 
97  real :: u_nonsym(szi_(g),szj_(g))
98  real :: v_nonsym(szi_(g),szj_(g))
99  real :: u_resym(szib_(g),szj_(g))
100  real :: v_resym(szi_(g),szjb_(g))
101  character(len=128) :: mesg2
102 
103  integer :: i, j, is_ch, ie_ch, js_ch, je_ch
104  integer :: isq, ieq, jsq, jeq, isd, ied, jsd, jed, isdb, iedb, jsdb, jedb
105  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
106  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
107  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
108 
109  if ((isd == isdb) .and. (jsd == jsdb)) return
110 
111  do i=isd,ied ; do j=jsd,jed
112  u_nonsym(i,j) = u_comp(i,j) ; v_nonsym(i,j) = v_comp(i,j)
113  enddo ; enddo
114 
115  if (.not.associated(g%Domain_aux)) call mom_error(fatal," check_redundant"//&
116  " called with a non-associated auxiliary domain the grid type.")
117  call pass_vector(u_nonsym, v_nonsym, g%Domain_aux, direction, stagger)
118 
119  do i=isdb,iedb ; do j=jsd,jed ; u_resym(i,j) = u_comp(i,j) ; enddo ; enddo
120  do i=isd,ied ; do j=jsdb,jedb ; v_resym(i,j) = v_comp(i,j) ; enddo ; enddo
121  do i=isd,ied ; do j=jsd,jed
122  u_resym(i,j) = u_nonsym(i,j) ; v_resym(i,j) = v_nonsym(i,j)
123  enddo ; enddo
124  call pass_vector(u_resym, v_resym, g%Domain, direction, stagger)
125 
126  is_ch = isq ; ie_ch = ieq ; js_ch = jsq ; je_ch = jeq
127  if (present(is)) is_ch = is ; if (present(ie)) ie_ch = ie
128  if (present(js)) js_ch = js ; if (present(js)) je_ch = je
129 
130  do i=is_ch,ie_ch ; do j=js_ch+1,je_ch
131  if (u_resym(i,j) /= u_comp(i,j)) then
132  write(mesg2,'(" redundant u-components",2(1pe12.4)," differ by ", &
133  & 1pe12.4," at i,j = ",2i4," on pe ",i4)') &
134  u_comp(i,j), u_resym(i,j),u_comp(i,j)-u_resym(i,j),i,j,pe_here()
135  write(*,'(A130)') trim(mesg)//trim(mesg2)
136  endif
137  enddo ; enddo
138  do i=is_ch+1,ie_ch ; do j=js_ch,je_ch
139  if (v_resym(i,j) /= v_comp(i,j)) then
140  write(mesg2,'(" redundant v-comps",2(1pe12.4)," differ by ", &
141  & 1pe12.4," at i,j = ",2i4," x,y = ",2(1pe12.4)" on pe ",i4)') &
142  v_comp(i,j), v_resym(i,j),v_comp(i,j)-v_resym(i,j),i,j, &
143  g%geoLonBu(i,j), g%geoLatBu(i,j), pe_here()
144  write(*,'(A155)') trim(mesg)//trim(mesg2)
145  endif
146  enddo ; enddo
