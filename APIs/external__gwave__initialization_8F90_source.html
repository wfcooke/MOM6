44 subroutine external_gwave_initialize_thickness(h, G, param_file)
45  real, intent(out), dimension(NIMEM_,NJMEM_, NKMEM_) :: h
46  type(ocean_grid_type), intent(in) :: g
47  type(param_file_type), intent(in) :: param_file
48 ! Arguments: h - The thickness that is being initialized.
49 ! (in) G - The ocean's grid structure.
50 ! (in) param_file - A structure indicating the open file to parse for
51 ! model parameter values.
52 
53 ! This subroutine initializes layer thicknesses for the external_gwave experiment
54  real :: e0(szk_(g)) ! The resting interface heights, in m, usually !
55  ! negative because it is positive upward. !
56  real :: e_pert(szk_(g)) ! Interface height perturbations, positive !
57  ! upward, in m. !
58  real :: eta1d(szk_(g)+1)! Interface height relative to the sea surface !
59  ! positive upward, in m. !
60  real :: ssh_anomaly_height ! Vertical height of ssh anomaly
61  real :: ssh_anomaly_width ! Lateral width of anomaly
62  character(len=40) :: mod = "external_gwave_initialize_thickness" ! This subroutine's name.
63  integer :: i, j, k, is, ie, js, je, nz
64  real :: pi, xnondim
65 
66  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
67 
68  call mom_mesg(" external_gwave_initialization.F90, external_gwave_initialize_thickness: setting thickness", 5)
69 
70  call get_param(param_file, mod, "SSH_ANOMALY_HEIGHT", ssh_anomaly_height, &
71  "The vertical displacement of the SSH anomaly. ", units="m", &
72  fail_if_missing=.true.)
73  call get_param(param_file, mod, "SSH_ANOMALY_WIDTH", ssh_anomaly_width, &
74  "The lateral width of the SSH anomaly. ", units="coordinate", &
75  fail_if_missing=.true.)
76  pi = 4.0*atan(1.0)
77  do j=g%jsc,g%jec ; do i=g%isc,g%iec
78  xnondim = (g%geoLonT(i,j)-g%west_lon-0.5*g%len_lon) / ssh_anomaly_width
79  xnondim = min(1., abs(xnondim))
80  eta1d(1) = ssh_anomaly_height * 0.5 * ( 1. + cos(pi*xnondim) ) ! Cosine bell
81  do k=2,nz
82  eta1d(k) = -g%max_depth & ! Stretch interior interfaces with SSH
83  + (eta1d(1)+g%max_depth) * ( real(nz+1-k)/real(nz) ) ! Stratification
84  enddo
85  eta1d(nz+1) = -g%max_depth ! Force bottom interface to bottom
86  do k=1,nz
87  h(i,j,k) = eta1d(k) - eta1d(k+1)
88  enddo
89  enddo ; enddo
