395 subroutine mom_thermovar_chksum(mesg, tv, G)
396  character(len=*), intent(in) :: mesg
397  type(thermo_var_ptrs), intent(in) :: tv
398  type(ocean_grid_type), intent(in) :: g
399 ! This subroutine writes out chksums for the model's basic state variables.
400 ! Arguments: mesg - A message that appears on the chksum lines.
401 ! (in) u - Zonal velocity, in m s-1.
402 ! (in) v - Meridional velocity, in m s-1.
403 ! (in) h - Layer thickness, in m.
404 ! (in) uh - Volume flux through zonal faces = u*h*dy, m3 s-1.
405 ! (in) vh - Volume flux through meridional faces = v*h*dx, in m3 s-1.
406 ! (in) G - The ocean's grid structure.
407  integer :: is, ie, js, je, nz
408  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
409 
410  ! Note that for the chksum calls to be useful for reproducing across PE
411  ! counts, there must be no redundant points, so all variables use is..ie
412  ! and js...je as their extent.
413  if (associated(tv%T)) &
414  call hchksum(tv%T, mesg//" tv%T",g)
415  if (associated(tv%S)) &
416  call hchksum(tv%S, mesg//" tv%S",g)
417  if (associated(tv%frazil)) &
418  call hchksum(tv%frazil, mesg//" tv%frazil",g)
419  if (associated(tv%salt_deficit)) &
420  call hchksum(tv%salt_deficit, mesg//" tv%salt_deficit",g)
421  if (associated(tv%TempxPmE)) &
422  call hchksum(tv%TempxPmE, mesg//" tv%TempxPmE",g)
423 end subroutine mom_thermovar_chksum
