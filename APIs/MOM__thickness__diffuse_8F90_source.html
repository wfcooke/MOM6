26 !* The subroutine in this file implements horizontal interface *
27 !* depth diffusion. In a Z-coordinate model, this would be described *
28 !* as Gent-McWilliams diffusion, with an identical interpretation *
29 !* of the diffusion coefficient. The coefficient is locally limited *
30 !* to guarantee numerical stability, and are vertically uniform. *
31 !* *
32 !* Macros written all in capital letters are defined in MOM_memory.h. *
33 !* *
34 !* A small fragment of the grid is shown below: *
35 !* *
36 !* j+1 x ^ x ^ x At x: q *
37 !* j+1 > o > o > At ^: v, vh, vav *
38 !* j x ^ x ^ x At >: u, uh, uav *
39 !* j > o > o > At o: h *
40 !* j-1 x ^ x ^ x *
41 !* i-1 i i+1 At x & ^: *
42 !* i i+1 At > & o: *
43 !* *
44 !* The boundaries always run through q grid points (x). *
45 !* *

118 ! (in) VarMix - A structure containing a number of fields related to

137  ! using density gradients at u points. The physically correct
138  ! slopes occur at 0, while 1 is used for numerical closures.
139  real, dimension(SZI_(G), SZJB_(G), SZK_(G)+1) :: &
140  kh_v, & ! The interface height diffusivities in v-columns, in m2 s-1.
141  int_slope_v ! A nondimensional ratio from 0 to 1 that gives the relative
142  ! weighting of the interface slopes to that calculated also
143  ! using density gradients at v points. The physically correct
144  ! slopes occur at 0, while 1 is used for numerical closures.

146  real :: kh_v_cfl(szi_(g), szjb_(g)) ! diffusivities at u & v grid points,
147  ! in m2 s-1.

164  endif

172  call find_eta(h, tv, g%g_Earth, g, e, halo_size=1)

174  ! Set the diffusivities.
175  do j=js,je ; do i=is-1,ie
176  khth_loc = cs%Khth
177  if (use_varmix) &
178  khth_loc = khth_loc + cs%KHTH_Slope_Cff*varmix%L2u(i,j)*varmix%SN_u(i,j)

189  endif

195  enddo ; enddo ; enddo
196 
197  do j=js-1,je ; do i=is,ie
198  khth_loc = cs%Khth
199  if (use_varmix) &
200  khth_loc = khth_loc + cs%KHTH_Slope_Cff*varmix%L2v(i,j)*varmix%SN_v(i,j)
201  if (ASSOCIATED(meke)) then
202  if (associated(meke%Kh)) &
203  khth_loc = khth_loc + meke%KhTh_fac*sqrt(meke%Kh(i,j)*meke%Kh(i,j+1))
204  endif
205  if (resoln_scaled) &
206  khth_loc = khth_loc * 0.5*(varmix%Res_fn_h(i,j) + varmix%Res_fn_h(i,j+1))
207  if (cs%Khth_Max > 0) then
208  khth_loc = max(cs%Khth_min, min(khth_loc,cs%Khth_Max))
209  else
210  khth_loc = max(cs%Khth_min, khth_loc)
211  endif

213  kh_v(i,j,1) = min(kh_v_cfl(i,j), khth_loc)
214  enddo ; enddo
215  do k=2,nz+1 ; do j=js-1,je ; do i=is,ie

227  if (cs%debug) then

229  call vchksum(kh_v(:,:,:),"Kh_v",g,haloshift=0)
230  call uchksum(int_slope_u(:,:,:),"int_slope_u",g,haloshift=0)

235

237  int_slope_u, int_slope_v)
238 
239  do k=1,nz
240  do j=js,je ; do i=is-1,ie
241  uhtr(i,j,k) = uhtr(i,j,k) + uhd(i,j,k)*dt
242  if (ASSOCIATED(cdp%uhGM)) cdp%uhGM(i,j,k) = uhd(i,j,k)
243  enddo ; enddo
244  do j=js-1,je ; do i=is,ie
245  vhtr(i,j,k) = vhtr(i,j,k) + vhd(i,j,k)*dt
246  if (ASSOCIATED(cdp%vhGM)) cdp%vhGM(i,j,k) = vhd(i,j,k)
247  enddo ; enddo
248  do j=js,je ; do i=is,ie
249  h(i,j,k) = h(i,j,k) - dt * g%IareaT(i,j) * &
250  ((uhd(i,j,k) - uhd(i-1,j,k)) + (vhd(i,j,k) - vhd(i,j-1,k)))
251  if (h(i,j,k) < g%Angstrom) h(i,j,k) = g%Angstrom
252  enddo ; enddo
253  enddo
254 
255  if (ASSOCIATED(meke) .and. ASSOCIATED(varmix)) then
256  if (ASSOCIATED(meke%Rd_dx_h) .and. ASSOCIATED(varmix%Rd_dx_h)) then
257  do j=js,je ; do i=is,ie
258  meke%Rd_dx_h(i,j) = varmix%Rd_dx_h(i,j)
259  enddo ; enddo
260  endif
261  endif
262 
263  if (cs%debug) then
264  call uchksum(uhd(:,:,:),"thickness_diffuse uhD",g,haloshift=0)
265  call vchksum(vhd(:,:,:),"thickness_diffuse vhD",g,haloshift=0)
266  call uchksum(uhtr(:,:,:),"thickness_diffuse uhtr",g,haloshift=0)
267  call vchksum(vhtr(:,:,:),"thickness_diffuse vhtr",g,haloshift=0)
268  call hchksum(h(:,:,:),"thickness_diffuse h",g,haloshift=0)
269  endif
270 
271 ! Offer diagnostic fields for averaging.
272  if (query_averaging_enabled(cs%diag)) then
273  if (cs%id_uhGM > 0) call post_data(cs%id_uhGM, cdp%uhGM, cs%diag)
274  if (cs%id_vhGM > 0) call post_data(cs%id_vhGM, cdp%vhGM, cs%diag)
275  if (cs%id_GMwork > 0) call post_data(cs%id_GMwork, cs%GMwork, cs%diag)
276  if (cs%id_KH_u > 0) call post_data(cs%id_KH_u, kh_u, cs%diag)
277  if (cs%id_KH_v > 0) call post_data(cs%id_KH_v, kh_v, cs%diag)
278  if (cs%id_KH_u1 > 0) call post_data(cs%id_KH_u1, kh_u(:,:,1), cs%diag)
279  if (cs%id_KH_v1 > 0) call post_data(cs%id_KH_v1, kh_v(:,:,1), cs%diag)
280  endif

768  ((e(i,j,k) + e(i,j,k+1)) + (e(i,j+1,k) + e(i+1,j,k+1))) )
769  endif
770  enddo ; enddo
771  enddo ! k-loop
772 
773  ! In layer 1, enforce the boundary conditions that Sfn(z=0) = 0.0
774  if (.not.find_work .or. .not.(use_eos)) then
775  do j=js,je ; do i=is-1,ie ; uhd(i,j,1) = -uhtot(i,j) ; enddo ; enddo
776  do j=js-1,je ; do i=is,ie ; vhd(i,j,1) = -vhtot(i,j) ; enddo ; enddo
777  else
778  if (use_eos) then ; do j=js,je
779  do i=is-1,ie
780  pres_u(i) = 0.5*(pres(i,j,1) + pres(i+1,j,1))
781  t_u(i) = 0.5*(t(i,j,1) + t(i+1,j,1))
782  s_u(i) = 0.5*(s(i,j,1) + s(i+1,j,1))
783  enddo
784  call calculate_density_derivs(t_u, s_u, pres_u, drho_dt_u(:,j), &
785  drho_ds_u(:,j), (is-isdb+1)-1, ie-is+2, tv%eqn_of_state)
786  enddo ; endif
787  do j=js,je ; do i=is-1,ie
788  uhd(i,j,1) = -uhtot(i,j)
789 
790  if (use_eos) then
791  drdib = drho_dt_u(i,j) * (t(i+1,j,1)-t(i,j,1)) + &
792  drho_ds_u(i,j) * (s(i+1,j,1)-s(i,j,1))
793  endif
794  work_u(i,j) = work_u(i,j) + g_scale * ( (uhd(i,j,1) * drdib) * 0.25 * &
795  ((e(i,j,1) + e(i,j,2)) + (e(i+1,j,1) + e(i+1,j,2))) )
796 
797  enddo ; enddo
798 
799  if (use_eos) then ; do j=js-1,je
800  do i=is,ie
801  pres_v(i) = 0.5*(pres(i,j,1) + pres(i,j+1,1))
802  t_v(i) = 0.5*(t(i,j,1) + t(i,j+1,1))
803  s_v(i) = 0.5*(s(i,j,1) + s(i,j+1,1))
804  enddo
805  call calculate_density_derivs(t_v, s_v, pres_v, drho_dt_v(:,j), &
806  drho_ds_v(:,j), is, ie-is+1, tv%eqn_of_state)
807  enddo ; endif
808  do j=js-1,je ; do i=is,ie
809  vhd(i,j,1) = -vhtot(i,j)
810 
811  if (use_eos) then
812  drdjb = drho_dt_v(i,j) * (t(i,j+1,1)-t(i,j,1)) + &
813  drho_ds_v(i,j) * (s(i,j+1,1)-s(i,j,1))
814  endif
815  work_v(i,j) = work_v(i,j) - g_scale * ( (vhd(i,j,1) * drdjb) * 0.25 * &
816  ((e(i,j,1) + e(i,j,2)) + (e(i,j+1,1) + e(i,j+1,2))) )
817  enddo ; enddo
818  endif
819 
820  if (find_work) then ; do j=js,je ; do i=is,ie
821  ! Note that the units of Work_v and Work_u are W, while Work_h is W m-2.
822  work_h = 0.5 * g%IareaT(i,j) * &
823  ((work_u(i-1,j) + work_u(i,j)) + (work_v(i,j-1) + work_v(i,j)))
824  if (ASSOCIATED(cs%GMwork)) cs%GMwork(i,j) = work_h
825  if (ASSOCIATED(meke)) then ; if (ASSOCIATED(meke%GM_src)) then
826  meke%GM_src(i,j) = meke%GM_src(i,j) + work_h
827  endif ; endif
828  enddo ; enddo ; endif
829 
830  ! if (CS%id_sfn_x > 0) call post_data(CS%id_sfn_x, sfn_x, CS%diag)
831  ! if (CS%id_sfn_y > 0) call post_data(CS%id_sfn_y, sfn_y, CS%diag)
832  ! if (CS%id_sfn_slope_x > 0) call post_data(CS%id_sfn_slope_x, sfn_slope_x, CS%diag)
833  ! if (CS%id_sfn_slope_y > 0) call post_data(CS%id_sfn_slope_y, sfn_slope_y, CS%diag)
834 
835 end subroutine thickness_diffuse_full

837 subroutine add_detangling_kh(h, e, Kh_u, Kh_v, KH_u_CFL, KH_v_CFL, tv, dt, G, CS, &
838  int_slope_u, int_slope_v)

840  real, dimension(NIMEM_,NJMEM_,NK_INTERFACE_), intent(in) :: e
841  real, dimension(NIMEMB_,NJMEM_,NK_INTERFACE_), intent(inout) :: kh_u
842  real, dimension(NIMEM_,NJMEMB_,NK_INTERFACE_), intent(inout) :: kh_v
843  real, dimension(NIMEMB_,NJMEM_), intent(in) :: kh_u_cfl
844  real, dimension(NIMEM_,NJMEMB_), intent(in) :: kh_v_cfl
845  type(thermo_var_ptrs), intent(in) :: tv
846  real, intent(in) :: dt
847  type(ocean_grid_type), intent(in) :: g
848  type(thickness_diffuse_cs), pointer :: cs
849  real, dimension(NIMEMB_,NJMEM_,NK_INTERFACE_), intent(inout) :: int_slope_u
850  real, dimension(NIMEM_,NJMEMB_,NK_INTERFACE_), intent(inout) :: int_slope_v
851 ! Arguments: h - Layer thickness, in m.
852 ! (in) e - Interface heights relative to mean sea level, in m.
853 ! (inout) Kh_u - Thickness diffusivity on interfaces at u points, in m2 s-1.
854 ! (inout) Kh_v - Thickness diffusivity on interfaces at v points, in m2 s-1.
855 ! (in) Kh_u_CFL - Maximum stable thickness diffusivity at u points, in m2 s-1.
856 ! (in) Kh_v_CFL - Maximum stable thickness diffusivity at v points, in m2 s-1.
857 ! (in) tv - A structure containing pointers to any available
858 ! thermodynamic fields. Absent fields have NULL ptrs.
859 ! (in) dt - Time increment in s.
860 ! (in) G - The ocean's grid structure.
861 ! (in) CS - The control structure returned by a previous call to
862 ! thickness_diffuse_init.
863 ! (inout) int_slope_u - Ratios that determine how much of the isopycnal
864 ! slopes are taken directly from the interface slopes without
865 ! consideration of density gradients.
866 ! (inout) int_slope_v - Ratios that determine how much of the isopycnal
867 ! slopes are taken directly from the interface slopes without
868 ! consideration of density gradients.
869  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
870  de_top ! The distances between the top of a layer and the top of the
871  ! region where the detangling is applied, in H.
872  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)) :: &
873  kh_lay_u ! The tentative interface height diffusivity for each layer at
874  ! u points, in m2 s-1.
875  real, dimension(SZI_(G),SZJB_(G),SZK_(G)) :: &
876  kh_lay_v ! The tentative interface height diffusivity for each layer at
877  ! v points, in m2 s-1.
878  real, dimension(SZI_(G),SZJ_(G)) :: &
879  de_bot ! The distances from the bottom of the region where the
880  ! detangling is applied, in H.
881  real :: h1, h2 ! The thinner and thicker surrounding thicknesses, in H,
882  ! with the thinner modified near the boundaries to mask out
883  ! thickness variations due to topography, etc.
884  real :: jag_rat ! The nondimensional jaggedness ratio for a layer, going
885  ! from 0 (smooth) to 1 (jagged). This is the difference
886  ! between the arithmetic and harmonic mean thicknesses
887  ! normalized by the arithmetic mean thickness.
888  real :: kh_scale ! A ratio by which Kh_u_CFL is scaled for maximally jagged
889  ! layers, nondim.
890  real :: kh_det ! The detangling diffusivity, in m2 s-1.
891  real :: h_neglect ! A thickness that is so small it is usually lost
892  ! in roundoff and can be neglected, in H.
893 
894  real :: i_sl ! The absolute value of the larger in magnitude of the slopes
895  ! above and below.
896  real :: rsl ! The ratio of the smaller magnitude slope to the larger
897  ! magnitude one, ND. 0 <= Rsl <1.
898  real :: irsl ! The (limited) inverse of Rsl, ND. 1 < IRsl <= 1e9.
899  real :: dh ! The thickness gradient divided by the damping timescale
900  ! and the ratio of the face length to the adjacent cell
901  ! areas for comparability with the diffusivities, in m2 s-1.
902  real :: adh ! The absolute value of dH, in m2 s-1.
903  real :: sign ! 1 or -1, with the same sign as the layer thickness gradient.
904  real :: sl_k ! The sign-corrected slope of the interface above, ND.
905  real :: sl_kp1 ! The sign-corrected slope of the interface below, ND.
906  real :: i_sl_k ! The (limited) inverse of sl_K, ND.
907  real :: i_sl_kp1 ! The (limited) inverse of sl_Kp1, ND.
908  real :: i_4t ! A quarter of inverse of the damping timescale, in s-1.
909  real :: fn_r ! A function of Rsl, such that Rsl < Fn_R < 1.
910  real :: denom, i_denom ! A denominator and its inverse, various units.
911  real :: kh_min ! A local floor on the diffusivity, in m2 s-1.
912  real :: kh_max ! A local ceiling on the diffusivity, in m2 s-1.
913  real :: wt1, wt2 ! Nondimensional weights.
914  ! Variables used only in testing code.
915  ! real, dimension(SZK_(G)) :: uh_here
916  ! real, dimension(SZK_(G)+1) :: Sfn
917  real :: dkh ! An increment in the diffusivity, in m2 s-1.
918 
919  real, dimension(SZIB_(G),SZK_(G)+1) :: &
920  kh_bg, & ! The background (floor) value of Kh, in m2 s-1.
921  kh, & ! The tentative value of Kh, in m2 s-1.
922  kh_detangle, & ! The detangling diffusivity that could be used, in m2 s-1.
923  kh_min_max_p, & ! The smallest ceiling that can be placed on Kh(I,K)
924  ! based on the value of Kh(I,K+1), in m2 s-1.
925  kh_min_max_m, & ! The smallest ceiling that can be placed on Kh(I,K)
926  ! based on the value of Kh(I,K-1), in m2 s-1.
927  ! The following are variables that define the relationships between
928  ! successive values of Kh.
929  ! Search for Kh that satisfy...
930  ! Kh(I,K) >= Kh_min_m(I,K)*Kh(I,K-1) + Kh0_min_m(I,K)
931  ! Kh(I,K) >= Kh_min_p(I,K)*Kh(I,K+1) + Kh0_min_p(I,K)
932  ! Kh(I,K) <= Kh_max_m(I,K)*Kh(I,K-1) + Kh0_max_m(I,K)
933  ! Kh(I,K) <= Kh_max_p(I,K)*Kh(I,K+1) + Kh0_max_p(I,K)
934  kh_min_m , & ! See above, ND.
935  kh0_min_m , & ! See above, in m2 s-1.
936  kh_max_m , & ! See above, ND.
937  kh0_max_m, & ! See above, in m2 s-1.
938  kh_min_p , & ! See above, ND.
939  kh0_min_p , & ! See above, in m2 s-1.
940  kh_max_p , & ! See above, ND.
941  kh0_max_p ! See above, in m2 s-1.
942  real, dimension(SZIB_(G)) :: &
943  kh_max_max ! The maximum diffusivity permitted in a column.
944  logical, dimension(SZIB_(G)) :: &
945  do_i ! If true, work on a column.
946  integer :: i, j, k, n, ish, jsh, is, ie, js, je, nz, k_top
947  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
948 
949  k_top = g%nk_rho_varies + 1
950  h_neglect = g%H_subroundoff
951  ! The 0.5 is because we are not using uniform weightings, but are
952  ! distributing the diffusivities more effectively (with wt1 & wt2), but this
953  ! means that the additions to a single interface can be up to twice as large.
954  kh_scale = 0.5
955  if (cs%detangle_time > dt) kh_scale = 0.5 * dt / cs%detangle_time
956 
957  do j=js-1,je+1 ; do i=is-1,ie+1
958  de_top(i,j,k_top) = 0.0 ; de_bot(i,j) = 0.0
959  enddo ; enddo
960  do k=k_top+1,nz ; do j=js-1,je+1 ; do i=is-1,ie+1
961  de_top(i,j,k) = de_top(i,j,k-1) + h(i,j,k-1)
962  enddo ; enddo ; enddo
963 
964  do j=js,je ; do i=is-1,ie
965  kh_lay_u(i,j,nz) = 0.0 ; kh_lay_u(i,j,k_top) = 0.0
966  enddo ; enddo
967  do j=js-1,je ; do i=is,ie
968  kh_lay_v(i,j,nz) = 0.0 ; kh_lay_v(i,j,k_top) = 0.0
969  enddo ; enddo
970 
971  do k=nz-1,k_top+1,-1
972  ! Find the diffusivities associated with each layer.
973  do j=js-1,je+1 ; do i=is-1,ie+1
974  de_bot(i,j) = de_bot(i,j) + h(i,j,k+1)
975  enddo ; enddo
976 
977  do j=js,je ; do i=is-1,ie ; if (g%mask2dCu(i,j) > 0.0) then
978  if (h(i,j,k) > h(i+1,j,k)) then
979  h2 = h(i,j,k)
980  h1 = max( h(i+1,j,k), h2 - min(de_bot(i+1,j), de_top(i+1,j,k)) )
981  else
982  h2 = h(i+1,j,k)
983  h1 = max( h(i,j,k), h2 - min(de_bot(i,j), de_top(i,j,k)) )
984  endif
985  jag_rat = (h2 - h1)**2 / (h2 + h1 + h_neglect)**2
986  kh_lay_u(i,j,k) = (kh_scale * kh_u_cfl(i,j)) * jag_rat**2
987  endif ; enddo ; enddo
988 
989  do j=js-1,je ; do i=is,ie ; if (g%mask2dCv(i,j) > 0.0) then
990  if (h(i,j,k) > h(i,j+1,k)) then
991  h2 = h(i,j,k)
992  h1 = max( h(i,j+1,k), h2 - min(de_bot(i,j+1), de_top(i,j+1,k)) )
993  else
994  h2 = h(i,j+1,k)
995  h1 = max( h(i,j,k), h2 - min(de_bot(i,j), de_top(i,j,k)) )
996  endif
997  jag_rat = (h2 - h1)**2 / (h2 + h1 + h_neglect)**2
998  kh_lay_v(i,j,k) = (kh_scale * kh_v_cfl(i,j)) * jag_rat**2
999  endif ; enddo ; enddo
1000  enddo
1001 
1002  ! Limit the diffusivities
1003 
1004  i_4t = kh_scale / (4.0*dt)
1005 
1006  do n=1,2
1007  if (n==1) then ; jsh = js ; ish = is-1
1008  else ; jsh = js-1 ; ish = is ; endif
1009 
1010  do j=jsh,je
1011 
1012  ! First, populate the diffusivities
1013  if (n==1) then ! This is a u-column.
1014  do i=ish,ie
1015  do_i(i) = (g%mask2dCu(i,j) > 0.0)
1016  kh_max_max(i) = kh_u_cfl(i,j)
1017  enddo
1018  do k=1,nz+1 ; do i=ish,ie
1019  kh_bg(i,k) = kh_u(i,j,k) ; kh(i,k) = kh_bg(i,k)
1020  kh_min_max_p(i,k) = kh_bg(i,k) ; kh_min_max_m(i,k) = kh_bg(i,k)
1021  kh_detangle(i,k) = 0.0
1022  enddo ; enddo
1023  else ! This is a v-column.
1024  do i=ish,ie
1025  do_i(i) = (g%mask2dCv(i,j) > 0.0) ; kh_max_max(i) = kh_v_cfl(i,j)
1026  enddo
1027  do k=1,nz+1 ; do i=ish,ie
1028  kh_bg(i,k) = kh_v(i,j,k) ; kh(i,k) = kh_bg(i,k)
1029  kh_min_max_p(i,k) = kh_bg(i,k) ; kh_min_max_m(i,k) = kh_bg(i,k)
1030  kh_detangle(i,k) = 0.0
1031  enddo ; enddo
1032  endif
1033 
1034  ! Determine the limits on the diffusivities.
1035  do k=k_top,nz ; do i=ish,ie ; if (do_i(i)) then
1036  if (n==1) then ! This is a u-column.
1037  dh = 0.0
1038  denom = ((g%IareaT(i+1,j) + g%IareaT(i,j))*g%dy_Cu(i,j))
1039  ! This expression uses differences in e in place of h for better
1040  ! consistency with the slopes.
1041  if (denom > 0.0) &
1042  dh = i_4t * ((e(i+1,j,k) - e(i+1,j,k+1)) - &
1043  (e(i,j,k) - e(i,j,k+1))) / denom
1044  ! dH = I_4t * (h(i+1,j,k) - h(i,j,k)) / denom
1045 
1046  adh = abs(dh)
1047  sign = 1.0 ; if (dh < 0) sign = -1.0
1048  sl_k = sign * (e(i+1,j,k)-e(i,j,k)) * g%IdxCu(i,j)
1049  sl_kp1 = sign * (e(i+1,j,k+1)-e(i,j,k+1)) * g%IdxCu(i,j)
1050 
1051  ! Add the incremental diffusivites to the surrounding interfaces.
1052  ! Adding more to the more steeply sloping layers (as below) makes
1053  ! the diffusivities more than twice as effective.
1054  denom = (sl_k**2 + sl_kp1**2)
1055  wt1 = 0.5 ; wt2 = 0.5
1056  if (denom > 0.0) then
1057  wt1 = sl_k**2 / denom ; wt2 = sl_kp1**2 / denom
1058  endif
1059  kh_detangle(i,k) = kh_detangle(i,k) + wt1*kh_lay_u(i,j,k)
1060  kh_detangle(i,k+1) = kh_detangle(i,k+1) + wt2*kh_lay_u(i,j,k)
1061  else ! This is a v-column.
1062  dh = 0.0
1063  denom = ((g%IareaT(i,j+1) + g%IareaT(i,j))*g%dx_Cv(i,j))
1064  if (denom > 0.0) &
1065  dh = i_4t * ((e(i,j+1,k) - e(i,j+1,k+1)) - &
1066  (e(i,j,k) - e(i,j,k+1))) / denom
1067  ! dH = I_4t * (h(i,j+1,k) - h(i,j,k)) / denom
1068 
1069  adh = abs(dh)
1070  sign = 1.0 ; if (dh < 0) sign = -1.0
1071  sl_k = sign * (e(i,j+1,k)-e(i,j,k)) * g%IdyCv(i,j)
1072  sl_kp1 = sign * (e(i,j+1,k+1)-e(i,j,k+1)) * g%IdyCv(i,j)
1073 
1074  ! Add the incremental diffusviites to the surrounding interfaces.
1075  ! Adding more to the more steeply sloping layers (as below) makes
1076  ! the diffusivities more than twice as effective.
1077  denom = (sl_k**2 + sl_kp1**2)
1078  wt1 = 0.5 ; wt2 = 0.5
1079  if (denom > 0.0) then
1080  wt1 = sl_k**2 / denom ; wt2 = sl_kp1**2 / denom
1081  endif
1082  kh_detangle(i,k) = kh_detangle(i,k) + wt1*kh_lay_v(i,j,k)
1083  kh_detangle(i,k+1) = kh_detangle(i,k+1) + wt2*kh_lay_v(i,j,k)
1084  endif
1085 
1086  if (adh == 0.0) then
1087  kh_min_m(i,k+1) = 1.0 ; kh0_min_m(i,k+1) = 0.0
1088  kh_max_m(i,k+1) = 1.0 ; kh0_max_m(i,k+1) = 0.0
1089  kh_min_p(i,k) = 1.0 ; kh0_min_p(i,k) = 0.0
1090  kh_max_p(i,k) = 1.0 ; kh0_max_p(i,k) = 0.0
1091  elseif (adh > 0.0) then
1092  if (sl_k <= sl_kp1) then
1093  ! This case should only arise from nonlinearities in the equation of state.
1094  ! Treat it as though dedx(K) = dedx(K+1) & dH = 0.
1095  kh_min_m(i,k+1) = 1.0 ; kh0_min_m(i,k+1) = 0.0
1096  kh_max_m(i,k+1) = 1.0 ; kh0_max_m(i,k+1) = 0.0
1097  kh_min_p(i,k) = 1.0 ; kh0_min_p(i,k) = 0.0
1098  kh_max_p(i,k) = 1.0 ; kh0_max_p(i,k) = 0.0
1099  elseif (sl_k <= 0.0) then ! Both slopes are opposite to dH
1100  i_sl = -1.0 / sl_kp1
1101  rsl = -sl_k * i_sl ! 0 <= Rsl < 1
1102  irsl = 1e9 ; if (rsl > 1e-9) irsl = 1.0/rsl ! 1 < IRsl <= 1e9
1103 
1104  fn_r = rsl
1105  if (kh_max_max(i) > 0) &
1106  fn_r = min(sqrt(rsl), rsl + (adh * i_sl) / kh_max_max(i))
1107 
1108  kh_min_m(i,k+1) = fn_r ; kh0_min_m(i,k+1) = 0.0
1109  kh_max_m(i,k+1) = rsl ; kh0_max_m(i,k+1) = adh * i_sl
1110  kh_min_p(i,k) = irsl ; kh0_min_p(i,k) = -adh * (i_sl*irsl)
1111  kh_max_p(i,k) = 1.0/(fn_r + 1.0e-30) ; kh0_max_p(i,k) = 0.0
1112  elseif (sl_kp1 < 0.0) then ! Opposite (nonzero) signs of slopes.
1113  i_sl_k = 1e18 ; if (sl_k > 1e-18) i_sl_k = 1.0 / sl_k
1114  i_sl_kp1 = 1e18 ; if (-sl_kp1 > 1e-18) i_sl_kp1 = -1.0 / sl_kp1
1115 
1116  kh_min_m(i,k+1) = 0.0 ; kh0_min_m(i,k+1) = 0.0
1117  kh_max_m(i,k+1) = - sl_k*i_sl_kp1 ; kh0_max_m(i,k+1) = adh*i_sl_kp1
1118  kh_min_p(i,k) = 0.0 ; kh0_min_p(i,k) = 0.0
1119  kh_max_p(i,k) = sl_kp1*i_sl_k ; kh0_max_p(i,k) = adh*i_sl_k
1120 
1121  ! This limit does not use the slope weighting so that potentially
1122  ! sharp gradients in diffusivities are not forced to occur.
1123  kh_max = adh / (sl_k - sl_kp1)
1124  kh_min_max_p(i,k) = max(kh_min_max_p(i,k), kh_max)
1125  kh_min_max_m(i,k+1) = max(kh_min_max_m(i,k+1), kh_max)
