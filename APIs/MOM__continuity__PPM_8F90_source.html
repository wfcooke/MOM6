26 !* This program contains the subroutine that advects layer *
27 !* thickness. The scheme here uses a Piecewise-Parabolic method with *
28 !* a positive definite limiter. *
29 !* *
30 !* Macros written all in capital letters are defined in MOM_memory.h. *
31 !* *
32 !* A small fragment of the grid is shown below: *
33 !* *
34 !* j+1 x ^ x ^ x At x: q *
35 !* j+1 > o > o > At ^: v, vh *
36 !* j x ^ x ^ x At >: u, uh *
37 !* j > o > o > At o: h, hin *
38 !* j-1 x ^ x ^ x *
39 !* i-1 i i+1 At x & ^: *
40 !* i i+1 At > & o: *
41 !* *
42 !* The boundaries always run through q grid points (x). *
43 !* *

595 subroutine zonal_flux_layer(u, h, hL, hR, uh, duhdu, visc_rem, dt, G, j, &
596  ish, ieh, do_i, vol_cfl)
597  real, dimension(NIMEMB_), intent(in) :: u, visc_rem
598  real, dimension(NIMEM_), intent(in) :: h, hl, hr
599  real, dimension(NIMEMB_), intent(inout) :: uh, duhdu
600  real, intent(in) :: dt
601  type(ocean_grid_type), intent(inout) :: g
602  integer, intent(in) :: j, ish, ieh
603  logical, dimension(NIMEMB_), intent(in) :: do_i
604  logical, intent(in) :: vol_cfl
605 ! This subroutines evaluates the zonal mass or volume fluxes in a layer.
606 !
607 ! Arguments: u - Zonal velocity, in m s-1.
608 ! (in) h - Layer thickness used to calculate the fluxes, in H.
609 ! (in) hL, hR - Left- and right- thicknesses in the reconstruction, in H.
610 ! (out) uh - The zonal mass or volume transport, in H m2 s-1.
611 ! (out) duhdu - The partial derivative of uh with u, in H m.
612 ! (in) dt - Time increment in s.
613 ! (in) G - The ocean's grid structure.
614 ! (in) visc_rem - Both the fraction of the momentum originally in a
615 ! layer that remains after a time-step of viscosity,
616 ! and the fraction of a time-step's worth of a
617 ! barotropic acceleration that a layer experiences
618 ! after viscosity is applied. Nondimensional between
619 ! 0 (at the bottom) and 1 (far above the bottom).
620 ! (in) j, ish, ieh - The index range to work on.
621 ! (in) do_i - A logical flag indiciating which I values to work on.
622 ! (in) vol_CFL - If true, rescale the ratio of face areas to the cell
623 ! areas when estimating the CFL number.
624  real :: cfl ! The CFL number based on the local velocity and grid spacing, ND.
625  real :: curv_3 ! A measure of the thickness curvature over a grid length,
626  ! with the same units as h_in.
627  real :: h_marg ! The marginal thickness of a flux, in H.
628  integer :: i
629 
630  do i=ish-1,ieh ; if (do_i(i)) then
631  ! Set new values of uh and duhdu.
632  if (u(i) > 0.0) then
633  if (vol_cfl) then ; cfl = (u(i) * dt) * (g%dy_Cu(i,j) * g%IareaT(i,j))
634  else ; cfl = u(i) * dt * g%IdxT(i,j) ; endif
635  curv_3 = hl(i) + hr(i) - 2.0*h(i)
636  uh(i) = g%dy_Cu(i,j) * u(i) * &
637  (hr(i) + cfl * (0.5*(hl(i) - hr(i)) + curv_3*(cfl - 1.5)))
638  h_marg = hr(i) + cfl * ((hl(i) - hr(i)) + 3.0*curv_3*(cfl - 1.0))
639  elseif (u(i) < 0.0) then
640  if (vol_cfl) then ; cfl = (-u(i) * dt) * (g%dy_Cu(i,j) * g%IareaT(i+1,j))
641  else ; cfl = -u(i) * dt * g%IdxT(i+1,j) ; endif
642  curv_3 = hl(i+1) + hr(i+1) - 2.0*h(i+1)
643  uh(i) = g%dy_Cu(i,j) * u(i) * &
644  (hl(i+1) + cfl * (0.5*(hr(i+1)-hl(i+1)) + curv_3*(cfl - 1.5)))
645  h_marg = hl(i+1) + cfl * ((hr(i+1)-hl(i+1)) + 3.0*curv_3*(cfl - 1.0))
646  else
647  uh(i) = 0.0
648  h_marg = 0.5 * (hl(i+1) + hr(i))
649  endif
650  duhdu(i) = g%dy_Cu(i,j) * h_marg * visc_rem(i)
651  endif ; enddo
652 
653 end subroutine zonal_flux_layer

655 subroutine zonal_face_thickness(u, h, hL, hR, h_u, dt, G, LB, vol_CFL, visc_rem_u)
656  real, dimension(NIMEMB_,NJMEM_,NKMEM_), intent(in) :: u
657  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h, hl, hr
658  real, dimension(NIMEMB_,NJMEM_,NKMEM_), intent(inout) :: h_u
659  real, intent(in) :: dt
660  type(ocean_grid_type), intent(inout) :: g
661  type(loop_bounds_type), intent(in) :: lb
662  logical, intent(in) :: vol_cfl
663  real, dimension(NIMEMB_,NJMEM_,NKMEM_), intent(in), optional :: visc_rem_u
664 ! This subroutines sets the effective interface thickness at each zonal
665 ! velocity point.
666 !
667 ! Arguments: u - Zonal velocity, in m s-1.
668 ! (in) h - Layer thickness used to calculate the fluxes, in H.
669 ! (in) hL, hR - Left- and right- thicknesses in the reconstruction, in H.
670 ! (out) h_u - The thickness at zonal faces, in H.
671 ! (in) dt - Time increment in s.
672 ! (in) G - The ocean's grid structure.
673 ! (in) LB - A structure with the active thickness loop bounds.
674 ! (in) vol_CFL - If true, rescale the ratio of face areas to the cell
675 ! areas when estimating the CFL number.
676 ! (in, opt) visc_rem_u - Both the fraction of the momentum originally in a
677 ! layer that remains after a time-step of viscosity,
678 ! and the fraction of a time-step's worth of a
679 ! barotropic acceleration that a layer experiences
680 ! after viscosity is applied. Nondimensional between
681 ! 0 (at the bottom) and 1 (far above the bottom).
682  real :: cfl ! The CFL number based on the local velocity and grid spacing, ND.
683  real :: curv_3 ! A measure of the thickness curvature over a grid length,
684  ! with the same units as h_in.
685 ! real :: h_avg ! The average thickness of a flux, in H.
686  real :: h_marg ! The marginal thickness of a flux, in H.
687  integer :: i, j, k, ish, ieh, jsh, jeh, nz
688  ish = lb%ish ; ieh = lb%ieh ; jsh = lb%jsh ; jeh = lb%jeh ; nz = g%ke
689 
690  do k=1,nz ; do j=jsh,jeh ; do i=ish-1,ieh
691  if (u(i,j,k) > 0.0) then
692  if (vol_cfl) then ; cfl = (u(i,j,k) * dt) * (g%dy_Cu(i,j) * g%IareaT(i,j))
693  else ; cfl = u(i,j,k) * dt * g%IdxT(i,j) ; endif
694  curv_3 = hl(i,j,k) + hr(i,j,k) - 2.0*h(i,j,k)
695 ! h_avg = hR(i,j,k) + CFL * (0.5*(hL(i,j,k) - hR(i,j,k)) + curv_3*(CFL - 1.5))
696  h_marg = hr(i,j,k) + cfl * ((hl(i,j,k) - hr(i,j,k)) + 3.0*curv_3*(cfl - 1.0))
697  elseif (u(i,j,k) < 0.0) then
698  if (vol_cfl) then ; cfl = (-u(i,j,k)*dt) * (g%dy_Cu(i,j) * g%IareaT(i+1,j))
699  else ; cfl = -u(i,j,k) * dt * g%IdxT(i+1,j) ; endif
700  curv_3 = hl(i+1,j,k) + hr(i+1,j,k) - 2.0*h(i+1,j,k)
701 ! h_avg = hL(i+1,j,k) + CFL * (0.5*(hR(i+1,j,k)-hL(i+1,j,k)) + curv_3*(CFL - 1.5))
702  h_marg = hl(i+1,j,k) + cfl * ((hr(i+1,j,k)-hl(i+1,j,k)) + &
703  3.0*curv_3*(cfl - 1.0))
704  else
705 ! h_avg = 0.5 * (hl(i+1,j,k) + hr(i,j,k))
706  ! The choice to use the arithmetic mean here is somewhat arbitrariy, but
707  ! it should be noted that hl(i+1,j,k) and hr(i,j,k) are usually the same.
708  h_marg = 0.5 * (hl(i+1,j,k) + hr(i,j,k))
709  ! h_marg = (2.0 * hl(i+1,j,k) * hr(i,j,k)) / &
710  ! (hl(i+1,j,k) + hr(i,j,k) + G%H_subroundoff)
711  endif
712  ! This could perhaps be replaced by h_avg.
713  h_u(i,j,k) = h_marg
714  if (present(visc_rem_u)) h_u(i,j,k) = h_u(i,j,k) * visc_rem_u(i,j,k)
715  enddo ; enddo ; enddo
716 
717 end subroutine zonal_face_thickness

719 subroutine zonal_flux_adjust(u, h_in, hL, hR, uhbt, uh_tot_0, duhdu_tot_0, &
720  du, du_max_cfl, du_min_cfl, dt, g, cs, visc_rem, &
721  j, ish, ieh, do_i_in, full_precision, uh_3d)
722  real, dimension(NIMEMB_,NJMEM_,NKMEM_), intent(in) :: u
723  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h_in, hl, hr
724  real, dimension(NIMEMB_,NKMEM_), intent(in) :: visc_rem
725  real, dimension(NIMEMB_), intent(in) :: uhbt
726  real, dimension(NIMEMB_), intent(in) :: du_max_cfl, du_min_cfl
727  real, dimension(NIMEMB_), intent(in) :: uh_tot_0, duhdu_tot_0

737 ! This subroutine returns the barotropic velocity adjustment that gives the
738 ! desired barotropic (layer-summed) transport.
739 !

741 ! (in) h_in - Layer thickness used to calculate the fluxes, in H.
742 ! (in) hL, hR - Left- and right- thicknesses in the reconstruction, in H.
743 ! (in) uhbt - Summed volume flux through zonal faces, in H m2 s-1.
744 ! (in) uh_tot_0 - The summed transport with 0 adjustment, in H m2 s-1.
745 ! (in) duhdu_tot_0 - The partial derivative of du_err with du at 0
746 ! adjustment, in H m.
747 ! (out) du - The barotropic velocity adjustment, in m s-1.
748 ! (in) du_max_CFL, du_min_CFL - The acceptable range of du, in m s-1.
749 ! (in) dt - Time increment in s.
750 ! (in) G - The ocean's grid structure.
751 ! (in) CS - The control structure returned by a previous call to
752 ! continuity_PPM_init.
753 ! (in) visc_rem - Both the fraction of the momentum originally in a
754 ! layer that remains after a time-step of viscosity,
755 ! and the fraction of a time-step's worth of a
756 ! barotropic acceleration that a layer experiences
757 ! after viscosity is applied. Nondimensional between

760 ! (in) do_I_in - A logical flag indiciating which I values to work on.

870  enddo
871  endif
872  enddo ! itt-loop

1964 subroutine ppm_limit_pos(h_in, h_L, h_R, h_min, iis, iie, jis, jie)
1965  real, dimension(NIMEM_,NJMEM_), intent(in) :: h_in
1966  real, dimension(NIMEM_,NJMEM_), intent(inout) :: h_l, h_r
1967  real, intent(in) :: h_min
1968  integer, intent(in) :: iis, iie, jis, jie
1969 ! This subroutine limits the left/right edge values of the PPM reconstruction
1970 ! to give a reconstruction that is positive-definite. Here this is
1971 ! reinterpreted as giving a constant thickness if the mean thickness is less
1972 ! than h_min, with a minimum of h_min otherwise.
1973 ! Arguments: h_in - thickness of layer (2D)
1974 ! (inout) h_L - left edge value (2D)
1975 ! (inout) h_R - right edge value (2D)
1976 ! (in) h_min - The minimum thickness that can be obtained by a
1977 ! concave parabolic fit.
1978 ! (in) iis, iie, jis, jie - Index range for computation.
1979 
1980 ! Local variables
1981  real :: curv, dh, scale
1982  character(len=256) :: mesg
1983  integer :: i,j
1984 
1985  do j=jis,jie ; do i=iis,iie
1986  ! This limiter prevents undershooting minima within the domain with
1987  ! values less than h_min.
1988  curv = 3.0*(h_l(i,j) + h_r(i,j) - 2.0*h_in(i,j))
1989  if (curv > 0.0) then ! Only minima are limited.
1990  dh = h_r(i,j) - h_l(i,j)
1991  if (abs(dh) < curv) then ! The parabola's minimum is within the cell.
1992  if (h_in(i,j) <= h_min) then
1993  h_l(i,j) = h_in(i,j) ; h_r(i,j) = h_in(i,j)
1994  elseif (12.0*curv*(h_in(i,j) - h_min) < (curv**2 + 3.0*dh**2)) then
1995  ! The minimum value is h_in - (curv^2 + 3*dh^2)/(12*curv), and must
1996  ! be limited in this case. 0 < scale < 1.
1997  scale = 12.0*curv*(h_in(i,j) - h_min) / (curv**2 + 3.0*dh**2)
1998  h_l(i,j) = h_in(i,j) + scale*(h_l(i,j) - h_in(i,j))
1999  h_r(i,j) = h_in(i,j) + scale*(h_r(i,j) - h_in(i,j))
2000  endif
2001  endif
2002  endif
2003  enddo ; enddo

2010  integer, intent(in) :: iis, iie, jis, jie
2011 ! This subroutine limits the left/right edge values of the PPM reconstruction
2012 ! according to the monotonic prescription of Colella and Woodward, 1984.
2013 ! Arguments: h_in - thickness of layer (2D)
2014 ! (inout) h_l - left edge value (2D)
