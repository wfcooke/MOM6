26 !* This file contains the driver routine which selects which *
27 !* continuity solver will be called, based on run-time input. *
28 !* *
29 !* A small fragment of the grid is shown below: *
30 !* *
31 !* j+1 x ^ x ^ x At x: q *
32 !* j+1 > o > o > At ^: v, vh *
33 !* j x ^ x ^ x At >: u, uh *
34 !* j > o > o > At o: h, hin *
35 !* j-1 x ^ x ^ x *
36 !* i-1 i i+1 At x & ^: *
37 !* i i+1 At > & o: *
38 !* *
39 !* The boundaries always run through q grid points (x). *
40 !* *

98  type(bt_cont_type), pointer, optional :: bt_cont
99 ! This subroutine time steps the layer thicknesses, using a monotonically
100 ! limit, directionally split PPM scheme, based on Lin (1994).
101 
102 ! Arguments: u - Zonal velocity, in m s-1.
103 ! (in) v - Meridional velocity, in m s-1.
104 ! (in) hin - Initial layer thickness, in m.
105 ! (out) h - Final layer thickness, in m.
106 ! (out) uh - Volume flux through zonal faces = u*h*dy, m3 s-1.
107 ! (out) vh - Volume flux through meridional faces = v*h*dx,
108 ! in m3 s-1.
109 ! (in) dt - Time increment in s.
110 ! (in) G - The ocean's grid structure.
111 ! (in) CS - The control structure returned by a previous call to
112 ! continuity_init.
113 ! (in, opt) uhbt - The summed volume flux through zonal faces, m3 s-1.
114 ! (in, opt) vhbt - The summed volume flux through meridional faces, m3 s-1.
115 ! (in, opt) OBC - This open boundary condition type specifies whether, where,
116 ! and what open boundary conditions are used.
117 ! (in, opt) visc_rem_u - Both the fraction of the momentum originally in a
118 ! (in, opt) visc_rem_v - layer that remains after a time-step of viscosity,
119 ! and the fraction of a time-step's worth of a
120 ! barotropic acceleration that a layer experiences
121 ! after viscosity is applied, in the zonal (_u) and
122 ! meridional (_v) directions. Nondimensional between
123 ! 0 (at the bottom) and 1 (far above the bottom).
124 ! (out, opt) u_cor - The zonal velocities that give uhbt as the depth-
125 ! integrated transport, in m s-1.
126 ! (out, opt) v_cor - The meridional velocities that give vhbt as the
127 ! depth-integrated transport, in m s-1.
128 ! (in, opt) uhbt_aux - A second set of summed volume fluxes through zonal
129 ! (in, opt) vhbt_aux - and meridional faces, both in m3 s-1.
130 ! (out, opt) u_cor_aux - The zonal and meridional velocities that give uhbt_aux
131 ! (out, opt) v_cor_aux - and vhbt_aux as the depth-integrated transports,
132 ! both in m s-1.
133 ! (out, opt) BT_cont - A structure with elements that describe the effective
134 ! open face areas as a function of barotropic flow.
135  if (present(visc_rem_u) .neqv. present(visc_rem_v)) call mom_error(fatal, &
136  "MOM_continuity: Either both visc_rem_u and visc_rem_v or neither"// &
137  " one must be present in call to continuity.")
138  if (present(u_cor) .neqv. present(v_cor)) call mom_error(fatal, &
139  "MOM_continuity: Either both u_cor and v_cor or neither"// &
140  " one must be present in call to continuity.")
141  if (present(uhbt_aux) .neqv. present(vhbt_aux)) call mom_error(fatal, &
142  "MOM_continuity: Either both uhbt_aux and uhbt_aux or neither"// &
143  " one must be present in call to continuity.")
144  if (present(u_cor_aux) .neqv. present(v_cor_aux)) call mom_error(fatal, &
145  "MOM_continuity: Either both u_cor_aux and v_cor_aux or neither"// &
146  " one must be present in call to continuity.")
147  if (present(u_cor_aux) .neqv. present(uhbt_aux)) call mom_error(fatal, &
148  "MOM_continuity: u_cor_aux can only be calculated if uhbt_aux is"// &
149  " provided, and uhbt_aux has no other purpose. Include both arguments"//&
150  " or neither.")
151 
152  if (cs%continuity_scheme == ppm_scheme) then
153  call continuity_ppm(u, v, hin, h, uh, vh, dt, g, cs%PPM_CSp, uhbt, vhbt, obc, &
154  visc_rem_u, visc_rem_v, u_cor, v_cor, &
155  uhbt_aux, vhbt_aux, u_cor_aux, v_cor_aux, bt_cont)
156  else
157  call mom_error(fatal, "continuity: Unrecognized value of continuity_scheme")
158  endif
159 
160 end subroutine continuity

162 subroutine continuity_init(Time, G, param_file, diag, CS)

168 ! Arguments: Time - The current model time.
169 ! (in) G - The ocean's grid structure.
170 ! (in) param_file - A structure indicating the open file to parse for
171 ! model parameter values.
172 ! (in) diag - A structure that is used to regulate diagnostic output.
173 ! (in/out) CS - A pointer that is set to point to the control structure
174 ! for this module

176 #include "version_variable.h"

182  return
183  endif
184  allocate(cs)
185 
186  ! Read all relevant parameters and write them to the model log.
187  call log_version(param_file, mod, version)
188  call get_param(param_file, mod, "CONTINUITY_SCHEME", tmpstr, &
189  "CONTINUITY_SCHEME selects the discretization for the \n"//&
190  "continuity solver. The only valid value currently is: \n"//&
191  "\t PPM - use a positive-definite (or monotonic) \n"//&
192  "\t piecewise parabolic reconstruction solver.", &
193  default=ppm_string)
194 
195  tmpstr = uppercase(tmpstr) ; cs%continuity_scheme = 0
196  select case (trim(tmpstr))
197  case (ppm_string) ; cs%continuity_scheme = ppm_scheme
198  case default
199  call mom_mesg('continuity_init: CONTINUITY_SCHEME ="'//trim(tmpstr)//'"', 0)
200  call mom_mesg("continuity_init: The only valid value is currently "// &
201  trim(ppm_string), 0)
202  call mom_error(fatal, "continuity_init: Unrecognized setting "// &
203  "#define CONTINUITY_SCHEME "//trim(tmpstr)//" found in input file.")
204  end select
205 
206  if (cs%continuity_scheme == ppm_scheme) then
207  call continuity_ppm_init(time, g, param_file, diag, cs%PPM_CSp)
208  endif
209
