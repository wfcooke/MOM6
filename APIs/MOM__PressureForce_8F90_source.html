23 !* *
24 !* By Robert Hallberg, April 1994 - June 2008 *
25 !* *
26 !* This file contains the subroutine that directs the model to use *
27 !* the code that determines the horizontal accelerations due to *
28 !* pressure gradients. The two options currently available are a *
29 !* traditional Montgomery potential form, and the analytic finite *
30 !* volume form described in Adcroft, Hallberg and Harrison, 2008, *
31 !* Ocean Modelling, 22, 106-113. *
32 !* *
33 !* PressureForce takes 9 arguments, which are described below. If *
34 !* a non-split time stepping scheme is used, the last three arguments *
35 !* are ignored. *
36 !* *
37 !* Macros written all in capital letters are defined in MOM_memory.h. *
38 !* *
39 !* A small fragment of the grid is shown below: *
40 !* *
41 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
42 !* j+1 > o > o > At ^: v, PFv *
43 !* j x ^ x ^ x At >: u, PFu *
44 !* j > o > o > At o: h, bathyT, M, e, p, pbce, T, S *
45 !* j-1 x ^ x ^ x *
46 !* i-1 i i+1 *
47 !* i i+1 *
48 !* *
49 !* The boundaries always run through q grid points (x). *
50 !* *
51 !********+*********+*********+*********+*********+*********+*********+**

79 subroutine pressureforce(h, tv, PFu, PFv, G, CS, ALE_CSp, p_atm, pbce, eta)
80  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
81  type(thermo_var_ptrs), intent(in) :: tv
82  real, dimension(NIMEMB_,NJMEM_,NKMEM_), intent(out) :: pfu
83  real, dimension(NIMEM_,NJMEMB_,NKMEM_), intent(out) :: pfv
84  type(ocean_grid_type), intent(in) :: g
85  type(pressureforce_cs), pointer :: cs
86  type(ale_cs), pointer :: ale_csp
87  real, dimension(:,:), optional, pointer :: p_atm
88  real, dimension(NIMEM_,NJMEM_,NKMEM_), optional, intent(out) :: pbce
89  real, dimension(NIMEM_,NJMEM_), optional, intent(out) :: eta
90 
91 ! This subroutine works as a temporary interface between the model and the
92 ! Boussinesq and non-Boussinesq pressure force routines.
93 ! Descriptions of the variables are in each of the routines called in the
94 ! following conditional block.
95 
96  if (cs%Analytic_FV_PGF) then
97  if (g%Boussinesq) then
98  call pressureforce_afv_bouss(h, tv, pfu, pfv, g, cs%PressureForce_AFV_CSp, &
99  ale_csp, p_atm, pbce, eta)
100  else
101  call pressureforce_afv_nonbouss(h, tv, pfu, pfv, g, cs%PressureForce_AFV_CSp, &
102  p_atm, pbce, eta)
103  endif
104  else
105  if (g%Boussinesq) then
106  call pressureforce_mont_bouss(h, tv, pfu, pfv, g, cs%PressureForce_Mont_CSp, &
107  p_atm, pbce, eta)
108  else
109  call pressureforce_mont_nonbouss(h, tv, pfu, pfv, g, cs%PressureForce_Mont_CSp, &
110  p_atm, pbce, eta)
111  endif
112  endif
113 
114 end subroutine pressureforce

116 subroutine pressureforce_init(Time, G, param_file, diag, CS, tides_CSp)
117  type(time_type), target, intent(in) :: time
118  type(ocean_grid_type), intent(in) :: g
119  type(param_file_type), intent(in) :: param_file
120  type(diag_ctrl), target, intent(inout) :: diag
121  type(pressureforce_cs), pointer :: cs
122  type(tidal_forcing_cs), optional, pointer :: tides_csp
123 ! Arguments: Time - The current model time.
124 ! (in) G - The ocean's grid structure.
125 ! (in) param_file - A structure indicating the open file to parse for
126 ! model parameter values.
127 ! (in) diag - A structure that is used to regulate diagnostic output.
128 ! (in/out) CS - A pointer that is set to point to the control structure
129 ! for this module.
130 ! (in) tides_CSp - a pointer to the control structure of the tide module.
131 
132 ! This include declares and sets the variable "version".
133 #include "version_variable.h"
134  character(len=40) :: mod = "MOM_PressureForce" ! This module's name.
135 
136  if (associated(cs)) then
137  call mom_error(warning, "PressureForce_init called with an associated "// &
138  "control structure.")
139  return
140  else ; allocate(cs) ; endif
141 
142  ! Read all relevant parameters and write them to the model log.
143  call log_version(param_file, mod, version)
144  call get_param(param_file, mod, "ANALYTIC_FV_PGF", cs%Analytic_FV_PGF, &
145  "If true the pressure gradient forces are calculated \n"//&
146  "with a finite volume form that analytically integrates \n"//&
147  "the equations of state in pressure to avoid any \n"//&
148  "possibility of numerical thermobaric instability, as \n"//&
149  "described in Adcroft et al., O. Mod. (2008).", default=.true.)
150 
151  if (cs%Analytic_FV_PGF) then
152  call pressureforce_afv_init(time, g, param_file, diag, &
153  cs%PressureForce_AFV_CSp, tides_csp)
154  else
155  call pressureforce_mont_init(time, g, param_file, diag, &
156  cs%PressureForce_Mont_CSp, tides_csp)
157  endif
158 
159 end subroutine pressureforce_init
