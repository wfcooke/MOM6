24 !* *
25 !* By Alistair Adcroft and Robert Hallberg, 2010-2012 *
26 !* *
27 !* This file contains code that does the time-stepping of the *
28 !* adiabatic dynamic core, in this case with a pseudo-second order *
29 !* Runge-Kutta time stepping scheme for the momentum and a forward- *
30 !* backward coupling between the momentum and continuity equations, *
31 !* but without any splitting between the baroclinic and barotropic *
32 !* modes. Apart from the lack of splitting, this is closely analogous *
33 !* to the split time stepping scheme, and efforts have been taken to *
34 !* ensure that for certain configurations (e.g., very short *
35 !* baroclinic time steps, a single barotropic step per baroclinic *
36 !* step, and particular choices about how to coupled the baroclinic *
37 !* and barotropic solves, the two solutions reproduce each other. *
38 !* Although this time stepping scheme is not very efficient with a *
39 !* large number of layers, it is valuable for verifying the proper *
40 !* behavior of the more complicated split time stepping scheme, and *
41 !* is not too inefficient for use with only a few layers. *
42 !* *
43 !* The subroutine step_MOM_dyn_unsplit_RK2 actually does the time *
44 !* stepping, while register_restarts_dyn_unsplit_RK2 sets the fields *
45 !* that are found in a full restart file with this scheme, and *
46 !* initialize_dyn_unsplit_RK2 initializes the cpu clocks that are * *
47 !* used in this module. For largely historical reasons, this module *
48 !* does not have its own control structure, but shares the same *
49 !* control structure with MOM.F90 and the other MOM_dynamics_... *
50 !* modules. *
51 !* *
52 !* Macros written all in capital letters are defined in MOM_memory.h. *
53 !* *
54 !* A small fragment of the grid is shown below: *
55 !* *
56 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
57 !* j+1 > o > o > At ^: v, PFv, CAv, vh, diffv, tauy, vbt, vhtr *
58 !* j x ^ x ^ x At >: u, PFu, CAu, uh, diffu, taux, ubt, uhtr *
59 !* j > o > o > At o: h, bathyT, eta, T, S, tr *
60 !* j-1 x ^ x ^ x *
61 !* i-1 i i+1 *
62 !* i i+1 *
63 !* *
64 !* The boundaries always run through q grid points (x). *
65 !* *

406  ((cs%PFv(i,j,k) + cs%CAv(i,j,k)) + cs%diffv(i,j,k)))
407  v_in(i,j,k) = g%mask2dCv(i,j) * (v_in(i,j,k) + dt * &
408  ((cs%PFv(i,j,k) + cs%CAv(i,j,k)) + cs%diffv(i,j,k)))
409  enddo ; enddo ; enddo

411 ! up[n] <- up* + dt d/dz visc d/dz up
412 ! u[n] <- u*[n] + dt d/dz visc d/dz u[n]
413  call cpu_clock_begin(id_clock_vertvisc)
414  call vertvisc_coef(up, vp, h_av, fluxes, visc, dt, g, cs%vertvisc_CSp)

417  call vertvisc_coef(u_in, v_in, h_av, fluxes, visc, dt, g, cs%vertvisc_CSp)
418  call vertvisc(u_in, v_in, h_av, fluxes, visc, dt, cs%OBC, cs%ADp, cs%CDp, g, &

420  call cpu_clock_end(id_clock_vertvisc)
421  call cpu_clock_begin(id_clock_pass)
422  call pass_vector(up, vp, g%Domain)
423  call pass_vector(u_in, v_in, g%Domain)
424  call cpu_clock_end(id_clock_pass)
425 
426 ! uh = up[n] * h[n] (up[n] might be extrapolated to damp GWs)
427 ! h[n+1] = h[n] + dt div . uh
428  call cpu_clock_begin(id_clock_continuity)
429  call continuity(up, vp, h_in, h_in, uh, vh, &
430  dt, g, cs%continuity_CSp, obc=cs%OBC)
431  call cpu_clock_end(id_clock_continuity)
432  call cpu_clock_begin(id_clock_pass)
433  call pass_var(h_in, g%Domain)
434  call pass_vector(uh, vh, g%Domain)
435  call cpu_clock_end(id_clock_pass)
436 
437 ! Accumulate mass flux for tracer transport
438  do k=1,nz
439  do j=js-2,je+2 ; do i=isq-2,ieq+2
440  uhtr(i,j,k) = uhtr(i,j,k) + dt*uh(i,j,k)
441  enddo ; enddo
442  do j=jsq-2,jeq+2 ; do i=is-2,ie+2
443  vhtr(i,j,k) = vhtr(i,j,k) + dt*vh(i,j,k)
444  enddo ; enddo
445  enddo
446 
447  if (cs%debug) then
448  call mom_state_chksum("Corrector", u_in, v_in, h_in, uh, vh, g)
449  call mom_accel_chksum("Corrector accel", cs%CAu, cs%CAv, cs%PFu, cs%PFv, &
450  cs%diffu, cs%diffv, g)
451  endif
452 
453  if (g%Boussinesq) then
454  do j=js,je ; do i=is,ie ; eta_av(i,j) = -g%bathyT(i,j) ; enddo ; enddo
455  else
456  do j=js,je ; do i=is,ie ; eta_av(i,j) = 0.0 ; enddo ; enddo
457  endif
458  do k=1,nz ; do j=js,je ; do i=is,ie
459  eta_av(i,j) = eta_av(i,j) + h_av(i,j,k)
460  enddo ; enddo ; enddo
461 
462  if (dyn_p_surf) deallocate(p_surf)
463 
464 ! Here various terms used in to update the momentum equations are
465 ! offered for averaging.
466  if (cs%id_PFu > 0) call post_data(cs%id_PFu, cs%PFu, cs%diag)
467  if (cs%id_PFv > 0) call post_data(cs%id_PFv, cs%PFv, cs%diag)
468  if (cs%id_CAu > 0) call post_data(cs%id_CAu, cs%CAu, cs%diag)
469  if (cs%id_CAv > 0) call post_data(cs%id_CAv, cs%CAv, cs%diag)
470 
471 ! Here the thickness fluxes are offered for averaging.
472  if (cs%id_uh > 0) call post_data(cs%id_uh, uh, cs%diag)
473  if (cs%id_vh > 0) call post_data(cs%id_vh, vh, cs%diag)
474 
475 end subroutine step_mom_dyn_unsplit_rk2
