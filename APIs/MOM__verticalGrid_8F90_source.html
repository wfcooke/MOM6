80 subroutine verticalgridinit( param_file, GV )
81 ! This routine initializes the verticalGrid_type structure (GV).
82 ! All memory is allocated but not necessarily set to meaningful values until later.
83  type(param_file_type), intent(in) :: param_file ! Parameter file handle/type
84  type(verticalgrid_type), pointer :: gv ! The container for vertical grid data
85 ! This include declares and sets the variable "version".
86 #include "version_variable.h"
87  integer :: nk
88  character(len=16) :: mod = 'MOM_verticalGrid'
89 
90  if (associated(gv)) call mom_error(fatal, &
91  'verticalGridInit: called with an associated GV pointer.')
92  allocate(gv)
93 
94  ! Read all relevant parameters and write them to the model log.
95  call log_version(param_file, mod, version, &
96  "Parameters providing information about the vertical grid.")
97  call get_param(param_file, "MOM", "G_EARTH", gv%g_Earth, &
98  "The gravitational acceleration of the Earth.", &
99  units="m s-2", default = 9.80)
100  call get_param(param_file, "MOM", "RHO_0", gv%Rho0, &
101  "The mean ocean density used with BOUSSINESQ true to \n"//&
102  "calculate accelerations and the mass for conservation \n"//&
103  "properties, or with BOUSSINSEQ false to convert some \n"//&
104  "parameters from vertical units of m to kg m-2.", &
105  units="kg m-3", default=1035.0)
106  call get_param(param_file, mod, "BOUSSINESQ", gv%Boussinesq, &
107  "If true, make the Boussinesq approximation.", default=.true.)
108  call get_param(param_file, mod, "ANGSTROM", gv%Angstrom_z, &
109  "The minumum layer thickness, usually one-Angstrom.", &
110  units="m", default=1.0e-10)
111  if (.not.gv%Boussinesq) &
112  call get_param(param_file, mod, "H_TO_KG_M2", gv%H_to_kg_m2,&
113  "A constant that translates thicknesses from the model's \n"//&
114  "internal units of thickness to kg m-2.", units="kg m-2 H-1", &
115  default=1.0)
116 #ifdef STATIC_MEMORY_
117  ! Here NK_ is a macro, while nk is a variable.
118  call get_param(param_file, mod, "NK", nk, &
119  "The number of model layers.", units="nondim", &
120  static_value=nk_)
121  if (nk /= nk_) call mom_error(fatal, "MOM_grid_init: " // &
122  "Mismatched number of layers NK_ between MOM_memory.h and param_file")
123 
124 #else
125  call get_param(param_file, mod, "NK", nk, &
126  "The number of model layers.", units="nondim", fail_if_missing=.true.)
127 #endif
128  gv%ke = nk
129 
130  if (gv%Boussinesq) then
131  gv%H_to_kg_m2 = gv%Rho0
132  gv%kg_m2_to_H = 1.0/gv%Rho0
133  gv%m_to_H = 1.0
134  gv%H_to_m = 1.0
135  gv%Angstrom = gv%Angstrom_z
136  else
137  gv%kg_m2_to_H = 1.0 / gv%H_to_kg_m2
138  gv%m_to_H = gv%Rho0 * gv%kg_m2_to_H
139  gv%H_to_m = gv%H_to_kg_m2 / gv%Rho0
140  gv%Angstrom = gv%Angstrom_z*1000.0*gv%kg_m2_to_H
141  endif
142  gv%H_subroundoff = 1e-20 * max(gv%Angstrom,gv%m_to_H*1e-17)
143  gv%H_to_Pa = gv%g_Earth * gv%H_to_kg_m2
144 
145 ! Log derivative values.
146  call log_param(param_file, "MOM_grid", "M to THICKNESS", gv%m_to_H)
147 
148  alloc_( gv%sInterface(nk+1) )
149  alloc_( gv%sLayer(nk) )
150 
151 end subroutine verticalgridinit

153 subroutine setverticalgridaxes( Rlay, GV )
154 ! This routine sets the coordinate data assuming "layer mode" of the isopycnal model.
155  ! Arguments
156  type(verticalgrid_type), pointer :: gv ! The container for vertical grid data
157  real, dimension(GV%ke), intent(in) :: rlay ! The layer target density
158  ! Local variables
159  integer :: nk
160 
161  nk = gv%ke
162 
163  gv%zAxisLongName = 'Target Potential Density'
164  gv%zAxisUnits = 'kg m-3'
165  gv%sLayer(:) = rlay(:)
166  gv%sInterface(1) = 1.5*rlay(1) - 0.5*rlay(2)
167  gv%sInterface(2:nk) = 0.5*( rlay(1:nk-1) + rlay(2:nk) )
168  gv%sInterface(nk+1) = 1.5*rlay(nk) - 0.5*rlay(nk-1)
169 
170 end subroutine setverticalgridaxes
