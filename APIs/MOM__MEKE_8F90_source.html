50 type, public :: meke_cs ; private
51  ! Parameters

80 subroutine step_forward_meke(MEKE, h, visc, dt, G, CS)
81  type(meke_type), pointer :: meke
82  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h
83  type(vertvisc_type), intent(in) :: visc
84  real, intent(in) :: dt
85  type(ocean_grid_type), intent(inout) :: g
86  type(meke_cs), pointer :: cs
87 
88 ! Arguments: MEKE - A structure with MEKE-related fields (intent in/out).
89 ! (in) h - Layer thickness, in m or kg m-2.
90 ! (in) visc - A structure containing vertical viscosities and related
91 ! fields.
92 ! (in) dt - The time step in s.
93 ! (in) G - The ocean's grid structure.
94 ! (in) CS - The control structure returned by a previous call to
95 ! MEKE_init.
96 
97 ! By A. Adcroft and R. Hallberg, 2006-2009.
98 ! This subroutine determines updates the Mesoscale Eddy Kinitic Energy
99 ! (MEKE).
100 
101  real, dimension(SZI_(G),SZJ_(G)) :: &
102  mass, & ! The total mass of the water column, in kg m-2.
103  i_mass, & ! The inverse of mass, in m2 kg-1.
104  src, & ! The sum of all MEKE sources, in m2 s-3.
105  meke_decay, & ! The MEKE decay timescale, in s-1.
106  meke_gm_src, & ! The MEKE source from thickness mixing, in m2 s-3.
107  meke_mom_src, & ! The MEKE source from momentum, in m2 s-3.
108  drag_rate_visc, &
109  drag_rate ! The MEKE spindown timescale due to bottom drag, in s-1.
110  real, dimension(SZIB_(G),SZJ_(G)) :: &
111  meke_uflux, & ! The zonal diffusive flux of MEKE, in kg m2 s-3.
112  kh_u, & ! The zonal diffusivity that is actually used, in m2 s-1.
113  drag_vel_u ! A (vertical) viscosity associated with bottom drag at
114  ! u-points, in m s-1.
115  real, dimension(SZIB_(G),SZJB_(G)) :: &
116  meke_vflux, & ! The meridional diffusive flux of MEKE, in kg m2 s-3.
117  kh_v, & ! The meridional diffusivity that is actually used, in m2 s-1.
118  drag_vel_v ! A (vertical) viscosity associated with bottom drag at
119  ! v-points, in m s-1.
120  real :: kh_here, inv_kh_max
121  real :: cdrag2
122  real :: mass_neglect ! A negligible mass, in kg m-2.
123  real :: ldamping ! The MEKE damping rate in s-1.
124  real :: rho0 ! A density used to convert mass to distance, in kg m-3.
125  real :: sdt ! dt to use locally (could be scaled to accelerate)
126  real :: sdt_damp ! dt for damping (sdt could be split).
127  integer :: i, j, k, is, ie, js, je, isq, ieq, jsq, jeq, nz
128  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
129  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
130 
131  if (.not.associated(cs)) call mom_error(fatal, &
132  "MOM_MEKE: Module must be initialized before it is used.")
133  if (.not.associated(meke)) call mom_error(fatal, &
134  "MOM_MEKE: MEKE must be initialized before it is used.")
135 
136  ! Only integerate the MEKE equations if MEKE is required.
137  if (associated(meke%MEKE)) then
138 
139  sdt = dt*cs%MEKE_dtScale ! Scaled dt to use for time-stepping
140  if (cs%id_src > 0) then ; do j=js,je ; do i=is,ie
141  src(i,j) = cs%MEKE_BGsrc
142  enddo ; enddo ; endif
143 
144  rho0 = g%H_to_kg_m2 * g%m_to_H
145  mass_neglect = g%H_to_kg_m2 * g%H_subroundoff
146 
147  do j=js-1,je+1 ; do i=is-1,ie+1 ; mass(i,j) = 0.0 ; enddo ; enddo
148  do k=1,nz ; do j=js-1,je+1 ; do i=is-1,ie+1
149  mass(i,j) = mass(i,j) + g%H_to_kg_m2 * h(i,j,k)
150  enddo ; enddo ; enddo
151  do j=js-1,je+1 ; do i=is-1,ie+1
152  i_mass(i,j) = 0.0
153  if (mass(i,j) > 0.0) i_mass(i,j) = 1.0 / mass(i,j)
154  if (mass(i,j) < 0.0) mass(i,j) = 0.0 ! Should be unneccesary?
155  enddo ; enddo
156 
157  if (associated(meke%mom_src)) then
158  do j=js,je ; do i=is,ie
159  meke%MEKE(i,j) = meke%MEKE(i,j) - i_mass(i,j) * &
160  (sdt*cs%MEKE_FrCoeff)*meke%mom_src(i,j)
161  enddo ; enddo
162  if (cs%id_src > 0) then
163  do j=js,je ; do i=is,ie
164  src(i,j) = src(i,j) - cs%MEKE_FrCoeff*i_mass(i,j)*meke%mom_src(i,j)
165  enddo ; enddo
166  endif
167  endif
168 
169  if (associated(meke%GM_src)) then
170  do j=js,je ; do i=is,ie
171  meke%MEKE(i,j) = meke%MEKE(i,j) - i_mass(i,j) * &
172  (sdt*cs%MEKE_GMcoeff)*meke%GM_src(i,j)
173  enddo ; enddo
174  if (cs%id_src > 0) then
175  do j=js,je ; do i=is,ie
176  src(i,j) = src(i,j) - cs%MEKE_GMcoeff*i_mass(i,j)*meke%GM_src(i,j)
177  enddo ; enddo
178  endif
179  endif
180 
181  do j=js,je ; do i=is,ie
182  meke%MEKE(i,j) = max(0.0,meke%MEKE(i,j) + (sdt*cs%MEKE_BGsrc)*g%mask2dT(i,j))
183  enddo ; enddo
184 
185  ! With a depth-dependent (and possibly strong) damping, it seems
186  ! advisable to use Strang-splitting between the damping and diffusion.
187  sdt_damp = sdt ; if (cs%MEKE_KH >= 0.0) sdt_damp = 0.5*sdt
188 
189  if (cs%visc_drag) then
190  do j=js,je ; do i=is-1,ie
191  drag_vel_u(i,j) = 0.0
192  if ((g%mask2dCu(i,j) > 0.0) .and. (visc%bbl_thick_u(i,j) > 0.0)) &
193  drag_vel_u(i,j) = visc%kv_bbl_u(i,j) / visc%bbl_thick_u(i,j)
194  enddo ; enddo
195  do j=js-1,je ; do i=is,ie
196  drag_vel_v(i,j) = 0.0
197  if ((g%mask2dCu(i,j) > 0.0) .and. (visc%bbl_thick_v(i,j) > 0.0)) &
198  drag_vel_v(i,j) = visc%kv_bbl_v(i,j) / visc%bbl_thick_v(i,j)
199  enddo ; enddo
200 
201  cdrag2 = cs%MEKE_Cd_scale * cs%cdrag**2
202  do j=js,je ; do i=is,ie
203  drag_rate_visc(i,j) = (0.25*g%IareaT(i,j) * &
204  ((g%areaCu(i-1,j)*drag_vel_u(i-1,j) + &
205  g%areaCu(i,j)*drag_vel_u(i,j)) + &
206  (g%areaCv(i,j-1)*drag_vel_v(i,j-1) + &
207  g%areaCv(i,j)*drag_vel_v(i,j)) ) )
208  drag_rate(i,j) = (rho0 * i_mass(i,j)) * &
209  sqrt(drag_rate_visc(i,j)**2 + 2.0*cdrag2 * max(0.0,meke%MEKE(i,j)))
210  enddo ; enddo
211  elseif (cs%MEKE_Cd_scale >= 0.0) then
212  do j=js,je ; do i=is,ie
213  drag_rate(i,j) = (rho0 * i_mass(i,j)) * &
214  (cs%cdrag*(2.0*sqrt(max(0.0,meke%MEKE(i,j)) + cs%MEKE_Uscale**2)))
215  enddo ; enddo
216  endif
217 
218  if (cs%MEKE_damping + cs%MEKE_Cd_scale > 0.0) then
219  do j=js,je ; do i=is,ie
220  ldamping = cs%MEKE_damping
221  if (cs%MEKE_Cd_scale > 0.0) &
222  ldamping = cs%MEKE_damping + cs%MEKE_Cd_scale * drag_rate(i,j)
223 
224  meke%MEKE(i,j) = meke%MEKE(i,j) / (1.0 + sdt_damp*ldamping)
225  meke_decay(i,j) = ldamping*g%mask2dT(i,j)
226  enddo ; enddo
227  endif
228 
229  if (cs%MEKE_KH >= 0.0) then
230  call cpu_clock_begin(id_clock_pass)
231  call pass_var(meke%MEKE, g%Domain)
232  call cpu_clock_end(id_clock_pass)
233 
234  ! ### More elaborate prescriptions for Kh could be used here.
235  kh_here = cs%MEKE_Kh
236  do j=js,je ; do i=is-1,ie
237  ! Limit Kh to avoid CFL violations.
238  if (associated(meke%Kh)) &
239  kh_here = cs%MEKE_Kh + cs%KhMEKE_Fac*0.5*(meke%Kh(i,j)+meke%Kh(i+1,j))
240  inv_kh_max = 2.0*sdt * ((g%dy_Cu(i,j)*g%IdxCu(i,j)) * &
241  max(g%IareaT(i,j),g%IareaT(i+1,j)))
242  if (kh_here*inv_kh_max > 0.25) kh_here = 0.25 / inv_kh_max
243  kh_u(i,j) = kh_here
244 
245  meke_uflux(i,j) = ((kh_here * (g%dy_Cu(i,j)*g%IdxCu(i,j))) * &
246  ((2.0*mass(i,j)*mass(i+1,j)) / ((mass(i,j)+mass(i+1,j)) + mass_neglect)) ) * &
247  (meke%MEKE(i,j) - meke%MEKE(i+1,j))
248  enddo ; enddo
249  do j=js-1,je ; do i=is,ie
250  if (associated(meke%Kh)) &
251  kh_here = cs%MEKE_Kh + cs%KhMEKE_Fac*0.5*(meke%Kh(i,j)+meke%Kh(i,j+1))
252  inv_kh_max = 2.0*sdt * ((g%dx_Cv(i,j)*g%IdyCv(i,j)) * &
253  max(g%IareaT(i,j),g%IareaT(i,j+1)))
254  if (kh_here*inv_kh_max > 0.25) kh_here = 0.25 / inv_kh_max
255  kh_v(i,j) = kh_here
256 
257  meke_vflux(i,j) = ((kh_here * (g%dx_Cv(i,j)*g%IdyCv(i,j))) * &
258  ((2.0*mass(i,j)*mass(i,j+1)) / ((mass(i,j)+mass(i,j+1)) + mass_neglect)) ) * &
259  (meke%MEKE(i,j) - meke%MEKE(i,j+1))
260  enddo ; enddo
261  do j=js,je ; do i=is,ie
262  meke%MEKE(i,j) = meke%MEKE(i,j) + (sdt*(g%IareaT(i,j)*i_mass(i,j))) * &
263  ((meke_uflux(i-1,j) - meke_uflux(i,j)) + &
264  (meke_vflux(i,j-1) - meke_vflux(i,j)))
265  enddo ; enddo
266 
267  if ((cs%MEKE_damping + cs%MEKE_Cd_scale > 0.0) .and. (sdt>sdt_damp)) then
268  ! Recalculate the drag rate, since MEKE has changed.
269  if (cs%visc_drag) then
270  do j=js,je ; do i=is,ie
271  drag_rate(i,j) = (rho0 * i_mass(i,j)) * &
272  sqrt(drag_rate_visc(i,j)**2 + cdrag2 * 2.0*max(0.0,meke%MEKE(i,j)))
273  enddo ; enddo
274  elseif (cs%MEKE_Cd_scale >= 0.0) then
275  do j=js,je ; do i=is,ie
276  drag_rate(i,j) = (rho0 * i_mass(i,j)) * &
277  (cs%cdrag*2.0*(sqrt(max(0.0,meke%MEKE(i,j)) + cs%MEKE_Uscale**2)))
278  enddo ; enddo
279  endif
280 
281  do j=js,je ; do i=is,ie
282  ldamping = cs%MEKE_damping
283  if (cs%MEKE_Cd_scale > 0.0) &
284  ldamping = cs%MEKE_damping + cs%MEKE_Cd_scale * drag_rate(i,j)
285 
286  meke%MEKE(i,j) = meke%MEKE(i,j) / (1.0 + sdt_damp*ldamping)
287  meke_decay(i,j) = 0.5 * g%mask2dT(i,j) * (meke_decay(i,j) + ldamping)
288  enddo ; enddo
289  endif
290  endif
291 
292  call cpu_clock_begin(id_clock_pass)
293  call pass_var(meke%MEKE, g%Domain)
294  call cpu_clock_end(id_clock_pass)
295 
296  if (cs%MEKE_KhCoeff>0.) then ; if (cs%Rd_as_max_scale) then
297  do j=js-1,je+1 ; do i=is-1,ie+1
298  meke%Kh(i,j) = (cs%MEKE_KhCoeff*sqrt(2.*max(0.,meke%MEKE(i,j))*g%areaT(i,j))) * &
299  min(meke%Rd_dx_h(i,j), 1.0)
300  enddo ; enddo
301  else
302  do j=js-1,je+1 ; do i=is-1,ie+1
303  meke%Kh(i,j) = cs%MEKE_KhCoeff*sqrt(2.*max(0.,meke%MEKE(i,j))*g%areaT(i,j))
304  enddo ; enddo
305  endif ; endif
306  call cpu_clock_begin(id_clock_pass)
307  call pass_var(meke%Kh, g%Domain)
308  call cpu_clock_end(id_clock_pass)
309 
310 ! Offer fields for averaging.
311  if (cs%id_MEKE>0) call post_data(cs%id_MEKE, meke%MEKE, cs%diag)
312  if (cs%id_Kh>0) call post_data(cs%id_Kh, meke%Kh, cs%diag)
313  if (cs%id_KhMEKE_u>0) call post_data(cs%id_KhMEKE_u, kh_u, cs%diag)
314  if (cs%id_KhMEKE_v>0) call post_data(cs%id_KhMEKE_v, kh_v, cs%diag)
315  if (cs%id_src>0) call post_data(cs%id_src, src, cs%diag)
316  if (cs%id_decay>0) call post_data(cs%id_decay, meke_decay, cs%diag)
317  if (cs%id_GM_src>0) then
318  call post_data(cs%id_GM_src, meke%GM_src, cs%diag)
319  endif
320  if (cs%id_mom_src>0) then
321  call post_data(cs%id_mom_src, meke%mom_src, cs%diag)
322  endif
323 
324 ! else ! if MEKE%MEKE
325 ! call MOM_error(FATAL, "MOM_MEKE: MEKE%MEKE is not associated!")
326  endif
327 
328 end subroutine step_forward_meke

332 subroutine meke_init(Time, G, param_file, diag, CS, MEKE)
333  type(time_type), intent(in) :: time
334  type(ocean_grid_type), intent(inout) :: g
335  type(param_file_type), intent(in) :: param_file
336  type(diag_ctrl), target, intent(inout) :: diag
337  type(meke_cs), pointer :: cs
338  type(meke_type), pointer :: meke
339 ! This subroutine allocates space for and claculates the static variables used
340 ! by this module. The metrics may be effectively 0, 1, or 2-D arrays,
341 ! while fields like the background viscosities are 2-D arrays.
342 ! ALLOC is a macro defined in MOM_memory.h for allocate or nothing with
343 ! static memory.
344 !
345 ! Arguments:
346 ! Arguments: Time - The current model time.
347 ! (in) G - The ocean's grid structure.
348 ! (in) param_file - A structure indicating the open file to parse for
349 ! model parameter values.
350 ! (in) diag - A structure that is used to regulate diagnostic output.
351 ! (in/out) CS - A pointer that is set to point to the control structure
352 ! for this module
353 ! (in/out) MEKE - A structure with MEKE-related fields (intent in/out due to
354 ! halo updates).
355 
356 ! Local variables
357  integer :: is, ie, js, je, isd, ied, jsd, jed, nz
358 ! This include declares and sets the variable "version".
359 #include "version_variable.h"
360  character(len=40) :: mod = "MOM_MEKE" ! This module's name.
361 
362  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
363  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
364 
365  if (.not. associated(meke)) then
366  call mom_error(warning, "MEKE_init called with NO associated "// &
367  "MEKE-type structure.")
368  return
369  endif
370  if (associated(cs)) then
371  call mom_error(warning, &
372  "MEKE_init called with an associated control structure.")
373  return
374  else ; allocate(cs) ; endif
375 
376  call mom_mesg("MEKE_init: reading parameters ", 5)
377 
378  cs%diag => diag
379 
380  ! Read all relevant parameters and write them to the model log.
381  call log_version(param_file, mod, version, "")
382  call get_param(param_file, mod, "MEKE_DAMPING", cs%MEKE_damping, &
383  "The local depth-indepented MEKE dissipation rate.", &
384  units="s-1", default=0.0)
385  call get_param(param_file, mod, "MEKE_CD_SCALE", cs%MEKE_Cd_scale, &
386  "A scaling for the bottom drag applied to MEKE. This \n"//&
387  "should be less than 1 to account for the surface \n"//&
388  "intensification of MEKE and the fraction of MEKE that \n"//&
389  "may be temporarily stored as potential energy.", &
390  units="nondim", default=0.0)
391  call get_param(param_file, mod, "MEKE_GMCOEFF", cs%MEKE_GMcoeff, &
392  "The efficiency of the conversion of potential energy \n"//&
393  "into MEKE by the thickness mixing parameterization. \n"//&
394  "If MEKE_GMCOEFF is negative, this conversion is not \n"//&
395  "used or calculated.", units="nondim", default=-1.0)
396  call get_param(param_file, mod, "MEKE_FRCOEFF", cs%MEKE_FrCoeff, &
397  "The efficiency of the conversion of mean energy into \n"//&
398  "MEKE. If MEKE_FRCOEFF is negative, this conversion \n"//&
399  "is not used or calculated.", units="nondim", default=-1.0)
400  call get_param(param_file, mod, "MEKE_BGSRC", cs%MEKE_BGsrc, &
401  "A background energy source for MEKE.", units="W kg-1", &
402  default=0.0)
403  call get_param(param_file, mod, "MEKE_KH", cs%MEKE_Kh, &
404  "A background lateral diffusivity of MEKE, or a \n"//&
405  "Use a negative value to not apply lateral diffusion to MEKE.", &
406  units="m2 s-1", default=-1.0)
407  call get_param(param_file, mod, "MEKE_DTSCALE", cs%MEKE_dtScale, &
408  "A scaling factor to accelerate the time evolution of MEKE.", &
409  units="nondim", default=1.0)
410  call get_param(param_file, mod, "MEKE_KHCOEFF", cs%MEKE_KhCoeff, &
411  "A scaling factor which is combined with the square root \n"//&
412  "of MEKE times the grid-cell area to give MEKE%Kh, or a \n"//&
413  "negative value not to calculate MEKE%Kh. \n"//&
414  "This factor must be >0 for MEKE to contribute to the \n"//&
415  "thickness/tracer mixing in the rest of the model. \n", &
416  units="nondim", default=-1.0)
417  call get_param(param_file, mod, "MEKE_USCALE", cs%MEKE_Uscale, &
418  "The background velocity that is combined with MEKE to \n"//&
419  "calculate the bottom drag.", units="m s-1", default=1.0)
420  call get_param(param_file, mod, "CDRAG", cs%cdrag, &
421  "CDRAG is the drag coefficient relating the magnitude of \n"//&
