68 ! Initialization of topography
69 !------------------------------------------------------------------------------
70 subroutine dome2d_initialize_topography ( D, G, param_file, max_depth )
71  ! Arguments
72  real, dimension(NIMEM_,NJMEM_), intent(out) :: d
73  type(ocean_grid_type), intent(in) :: g
74  type(param_file_type), intent(in) :: param_file
75  real, intent(in) :: max_depth
76 
77  ! Local variables
78  integer :: i, j
79  real :: x, bay_depth, l1, l2
80 
81  ! location where downslope starts
82  l1 = dome2d_width_bay
83 
84  ! location where downslope reaches maximum depth
85  l2 = 1.0 - dome2d_width_bottom
86 
87  bay_depth = dome2d_depth_bay
88 
89  do i=g%isc,g%iec
90  do j=g%jsc,g%jec
91 
92  ! Compute normalized zonal coordinate
93  x = g%geoLonT(i,j) / g%len_lon;
94 
95  if ( x .le. l1 ) then
96  d(i,j) = bay_depth * max_depth
97  else if (( x .gt. l1 ) .and. ( x .lt. l2 )) then
98  d(i,j) = bay_depth * max_depth + (1.0-bay_depth) * max_depth * &
99  ( x - l1 ) / (l2 - l1)
100  else
101  d(i,j) = max_depth
102  end if
103 
104  enddo
105  enddo
106 end subroutine dome2d_initialize_topography

109 ! Initialization of thicknesses
110 !------------------------------------------------------------------------------
111 subroutine dome2d_initialize_thickness ( h, G, param_file )
112 
113  real, intent(out), dimension(NIMEM_,NJMEM_, NKMEM_) :: h
114  type(ocean_grid_type), intent(in) :: g
115  type(param_file_type), intent(in) :: param_file
116 
117 ! Arguments: h - The thickness that is being initialized.
118 ! (in) G - The ocean's grid structure.
119 ! (in) param_file - A structure indicating the open file to parse for
120 ! model parameter values.
121 
122 ! This subroutine initializes the layer thicknesses to be uniform.
123  real :: e0(szk_(g)) ! The resting interface heights, in m, usually !
124  ! negative because it is positive upward. !
125  real :: eta1d(szk_(g)+1)! Interface height relative to the sea surface !
126  ! positive upward, in m. !
127  integer :: i, j, k, is, ie, js, je, nz
128  real :: x
129  real :: delta_h
130  real :: min_thickness
131  character(len=40) :: verticalcoordinate
132 
133  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
134 
135  call mom_mesg("MOM_initialization.F90, initialize_thickness_uniform: setting thickness")
136 
137  call get_param(param_file,mod,"MIN_THICKNESS",min_thickness,'Minimum layer thickness',units='m',default=1.e-3)
138  call get_param(param_file,mod,"REGRIDDING_COORDINATE_MODE", verticalcoordinate, &
139  default=default_coordinate_mode)
140 
141  ! WARNING: this routine specifies the interface heights so that the last layer
142  ! is vanished, even at maximum depth. In order to have a uniform
143  ! layer distribution, use this line of code within the loop:
144  ! e0(k) = -G%max_depth * real(k-1) / real(nz)
145  ! To obtain a thickness distribution where the last layer is
146  ! vanished and the other thicknesses uniformly distributed, use:
147  ! e0(k) = -G%max_depth * real(k-1) / real(nz-1)
148  do k=1,nz
149  e0(k) = -g%max_depth * real(k-1) / real(nz)
150  enddo
151 
152  select case ( coordinatemode(verticalcoordinate) )
153 
154  case ( regridding_layer, regridding_rho )
155 
156  do j=js,je ; do i=is,ie
157  eta1d(nz+1) = -1.0*g%bathyT(i,j)
158  do k=nz,1,-1
159  eta1d(k) = e0(k)
160  if (eta1d(k) < (eta1d(k+1) + g%Angstrom_z)) then
161  eta1d(k) = eta1d(k+1) + g%Angstrom_z
162  h(i,j,k) = g%Angstrom_z
163  else
164  h(i,j,k) = eta1d(k) - eta1d(k+1)
165  endif
166  enddo
167 
168  x = g%geoLonT(i,j) / g%len_lon;
169  if ( x .le. dome2d_width_bay ) then
170  h(i,j,1:nz-1) = g%Angstrom;
171  h(i,j,nz) = dome2d_depth_bay * g%max_depth - (nz-1) * g%Angstrom;
172  end if
173 
174  end do ; end do
175 
176  ! case ( IC_RHO_C )
177  !
178  ! do j=js,je ; do i=is,ie
179  ! eta1D(nz+1) = -1.0*G%bathyT(i,j)
180  ! do k=nz,1,-1
181  ! eta1D(k) = e0(k)
182  ! if (eta1D(k) < (eta1D(k+1) + min_thickness)) then
183  ! eta1D(k) = eta1D(k+1) + min_thickness
184  ! h(i,j,k) = min_thickness
185  ! else
186  ! h(i,j,k) = eta1D(k) - eta1D(k+1)
187  ! endif
188  ! enddo
189  !
190  ! x = G%geoLonT(i,j) / G%len_lon;
191  ! if ( x .le. dome2d_width_bay ) then
192  ! h(i,j,1:nz-1) = min_thickness;
193  ! h(i,j,nz) = dome2d_depth_bay * G%max_depth - (nz-1) * min_thickness;
194  ! end if
195  !
196  ! enddo ; enddo
197 
198  case ( regridding_zstar )
199 
200  do j=js,je ; do i=is,ie
201  eta1d(nz+1) = -1.0*g%bathyT(i,j)
202  do k=nz,1,-1
203  eta1d(k) = e0(k)
204  if (eta1d(k) < (eta1d(k+1) + min_thickness)) then
205  eta1d(k) = eta1d(k+1) + min_thickness
206  h(i,j,k) = min_thickness
207  else
208  h(i,j,k) = eta1d(k) - eta1d(k+1)
209  endif
210  enddo
211  enddo ; enddo
212 
213  case ( regridding_sigma )
214  do j=js,je ; do i=is,ie
215  delta_h = g%bathyT(i,j) / nz;
216  h(i,j,:) = delta_h;
217  end do ; end do
218 
219  case default
220  call mom_error(fatal,"dome2d_initialize: "// &
221  "Unrecognized i.c. setup - set REGRIDDING_COORDINATE_MODE")
222 
223  end select
224 
225 end subroutine dome2d_initialize_thickness

229 ! Initialization of temperature and salinity
230 !------------------------------------------------------------------------------
231 subroutine dome2d_initialize_temperature_salinity ( T, S, h, G, param_file, &
232  eqn_of_state)
233  real, dimension(NIMEM_,NJMEM_, NKMEM_), intent(out) :: t, s
234  real, intent(in), dimension(NIMEM_,NJMEM_, NKMEM_) :: h
235  type(ocean_grid_type), intent(in) :: g
236  type(param_file_type), intent(in) :: param_file
237  type(eos_type), pointer :: eqn_of_state
238 
239  integer :: i, j, k, is, ie, js, je, nz
240  real :: x;
241  integer :: index_bay_z;
242  real :: delta_s, delta_t;
243  real :: s_ref, t_ref; ! Reference salinity and temerature within
244  ! surface layer
245  real :: s_range, t_range; ! Range of salinities and temperatures over the
246  ! vertical
247  real :: xi0, xi1;
248  character(len=40) :: verticalcoordinate
249 
250  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
251 
252  call get_param(param_file,mod,"REGRIDDING_COORDINATE_MODE", verticalcoordinate, &
253  default=default_coordinate_mode)
254  call get_param(param_file,mod,"S_REF",s_ref,'Reference salinity',units='1e-3',fail_if_missing=.true.)
255  call get_param(param_file,mod,"T_REF",t_ref,'Refernce temperature',units='C',fail_if_missing=.true.)
256  call get_param(param_file,mod,"S_RANGE",s_range,'Initial salinity range',units='1e-3',default=2.0)
257  call get_param(param_file,mod,"T_RANGE",t_range,'Initial temperature range',units='1e-3',default=0.0)
258 
259  t(:,:,:) = 0.0
260  s(:,:,:) = 0.0
261 
262  ! Linear salinity profile
263 
264  select case ( coordinatemode(verticalcoordinate) )
265 
266  case ( regridding_zstar, regridding_sigma )
267 
268  do j=js,je ; do i=is,ie
269  xi0 = 0.0;
270  do k = 1,nz
271  xi1 = xi0 + h(i,j,k) / g%max_depth;
272  s(i,j,k) = 34.0 + 0.5 * s_range * (xi0 + xi1);
273  xi0 = xi1;
274  enddo
275  enddo ; enddo
276 
277  case ( regridding_rho )
278 
279  do j=js,je ; do i=is,ie
280  xi0 = 0.0;
281  do k = 1,nz
282  xi1 = xi0 + h(i,j,k) / g%max_depth;
283  s(i,j,k) = 34.0 + 0.5 * s_range * (xi0 + xi1);
284  xi0 = xi1;
285  enddo
286  x = g%geoLonT(i,j) / g%len_lon;
287  if ( x .le. dome2d_width_bay ) then
288  s(i,j,nz) = 34.0 + s_range;
289  end if
290  enddo ; enddo
291 
292  case ( regridding_layer )
293 
294  delta_s = s_range / ( g%ke - 1.0 );
295  s(:,:,1) = s_ref;
296  do k = 2,g%ke
297  s(:,:,k) = s(:,:,k-1) + delta_s;
298  end do
299 
300  case default
301  call mom_error(fatal,"dome2d_initialize: "// &
302  "Unrecognized i.c. setup - set REGRIDDING_COORDINATE_MODE")
303 
304  end select
305 
306  ! Modify salinity and temperature when z coordinates are used
307  if ( coordinatemode(verticalcoordinate) .eq. regridding_zstar ) then
308  index_bay_z = nint( dome2d_depth_bay * g%ke );
309  do j = g%jsc,g%jec ; do i = g%isc,g%iec
310  x = g%geoLonT(i,j) / g%len_lon;
311  if ( x .le. dome2d_width_bay ) then
312  s(i,j,1:index_bay_z) = s_ref + s_range; ! Use for z coordinates
313  t(i,j,1:index_bay_z) = 1.0; ! Use for z coordinates
314  end if
315  end do ; end do ! i and j loops
316  end if ! Z initial conditions
317 
318  ! Modify salinity and temperature when sigma coordinates are used
319  if ( coordinatemode(verticalcoordinate) .eq. regridding_sigma ) then
320  do i = g%isc,g%iec ; do j = g%jsc,g%jec
321  x = g%geoLonT(i,j) / g%len_lon;
322  if ( x .le. dome2d_width_bay ) then
323  s(i,j,1:g%ke) = s_ref + s_range; ! Use for sigma coordinates
324  t(i,j,1:g%ke) = 1.0; ! Use for sigma coordinates
325  end if
326  end do ; end do
327  end if
328 
329  ! Modify temperature when rho coordinates are used
330  t(g%isc:g%iec,g%jsc:g%jec,1:g%ke) = 0.0;
331  if (( coordinatemode(verticalcoordinate) .eq. regridding_rho ) .or. ( coordinatemode(verticalcoordinate) .eq. regridding_layer )) then
332  do i = g%isc,g%iec ; do j = g%jsc,g%jec
333  x = g%geoLonT(i,j) / g%len_lon;
334  if ( x .le. dome2d_width_bay ) then
335  t(i,j,g%ke) = 1.0;
336  end if
337  end do ; end do
338  end if
339 
340 
341 end subroutine dome2d_initialize_temperature_salinity
